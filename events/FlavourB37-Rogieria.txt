#Dynasty Naming
country_event = {
	id = flavor_rogieria.1
	title = dynasty_setup.1.t
	desc = dynasty_setup.1.d
	picture = {
		trigger = {
			NOT = { has_dlc = "Rights of Man" }
		}
		picture = COURT_eventPicture
	}
	picture = {
		trigger = {
			has_dlc = "Rights of Man"
		}
		picture = ROYAL_COUPLE_FUTURE_eventPicture
	}
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		tag = B37
	}
	
	#Silmuna
	option = {		
		name = "flavor_rogieria.1.a"
		ai_chance = { factor = 98 }	
		if = {
			limit = {
				OR = {
					has_ruler = "Rogier Silmuna"
					has_ruler = "Rogier I Silmuna"
				}
			}
			define_heir = {
				dynasty = "Silmuna"
				age = 16
				male = yes
				adm = 1
				dip = 1
				mil = 1
				hide_skills = yes
			}
		}
		else = {
			define_ruler = {
				dynasty = "Silmuna"
				age = 16
				male = yes
				adm = 1
				dip = 1
				mil = 1
				hide_skills = yes
			}
		}
	}
	
	##New Dynasty Name after country
	option = {		
		name = "flavor_rogieria.1.b"
		ai_chance = { factor = 1 }	
		define_heir = {
			dynasty = "síl Rogieria"	#of the crossing
			age = 16
			male = yes
			adm = 1
			dip = 1
			mil = 1
			hide_skills = yes
		}
	}
	
	#Keep their name
	option = {		
		trigger = { NOT = { dynasty = "Silmuna"} }
		name = "flavor_rogieria.1.c"
		ai_chance = { factor = 1 }	
		define_heir = {
			dynasty = ROOT
			age = 16
			male = yes
			adm = 1
			dip = 1
			mil = 1
			hide_skills = yes
		}
	}
}

#Culture Shift
country_event = {
	id = flavor_rogieria.2
	title = flavor_luciande.2.t
	desc = flavor_luciande.2.d
	picture = EUROPEAN_REFUGEES_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = B37
		NOT = { has_country_modifier = legacy_of_adventurers_modifier }
		
		num_of_cities = 5
	}
	
	mean_time_to_happen = {
		months = 800
		modifier = {
			factor = 0.7
			stability = 1
		}
		modifier = {
			factor = 0.5
			num_of_cities = 10
		}
		modifier = {
			factor = 0.2
			in_golden_age = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					culture = ROOT
					any_owned_province = {
						superregion = escann_superregion
						region = forlorn_vale_region
					}
				}
				random_owned_province = {
					limit = {
						OR = {
							superregion = escann_superregion
							region = forlorn_vale_region
						}
						culture = ROOT
					}
					save_event_target_as = culture_shift_province
				}
			}
		}
	}
	
	# Birth of a New People
	option = {		
		name = flavor_luciande.1.a
		ai_chance = { factor = 65 }	
		
		event_target:culture_shift_province = {
			area = {
				limit = {
					owned_by = ROOT
					culture = ROOT
				}
				change_culture = rogieran
			}
		}
		
		add_country_modifier = {
			name = birth_of_a_new_people
			duration = 3650
		}
		
		change_primary_culture = rogieran
		
		if = {
			ruler_culture = culture_shift_province
			set_ruler_culture = rogieran
		}
		if = {
			heir_culture = culture_shift_province
			set_heir_culture = rogieran
		}
		if = {
			consort_culture = culture_shift_province
			set_consort_culture = rogieran
		}
	}
	
	option = { # Reject Freedom
		name = flavor_luciande.1.b
		trigger = {
			culture = ROOT
			ai = no
		}
		event_target:culture_shift_province = {
			change_culture = rogieran
		}
	}
}
