country_event = {
    id = flavor_bhuvauri.2
    title =  flavor_bhuvauri.2.t
    desc =  flavor_bhuvauri.2.d
    picture = ELECTION_REPUBLICAN_eventPicture
    
    is_triggered_only = yes
    
    trigger = {
        has_reform = free_slave_reform
        NOT = { has_country_flag = in_bhuvauri_election }
    }
    
    immediate = {
        hidden_effect = {
            set_country_flag = in_bhuvauri_election
        }
    }
	
	after = {
		hidden_effect = {
            clr_country_flag = in_bhuvauri_election
        }
	}
	
	#Dakinshi - Military Candidate
	option = {
		trigger = { NOT = { has_ruler_flag = bhuvauri_dakinshi_ruler } }
		name = flavor_bhuvauri.2.a
		if = {
			limit = { republic_with_female_rulers_trigger = yes }
			define_ruler = {
				max_random_mil = 3
				max_random_adm = 2
				max_random_dip = 1
				change_adm = 2
				change_dip = 1
				change_mil = 2
				culture = dakinshi
				religion = beast_memory
				random_gender = yes # Allows random gender for republics
			}
		}
		else = {
			define_ruler = {
				max_random_mil = 3
				max_random_adm = 2
				max_random_dip = 1
				change_adm = 2
				change_dip = 1
				change_mil = 2
				culture = dakinshi
				religion = beast_memory
			}
		}
		hidden_effect = {
			clr_ruler_flag = bhuvauri_bulwari_ruler
			set_ruler_flag = bhuvauri_dakinshi_ruler
			clr_ruler_flag = bhuvauri_fangaulan_ruler
		}
		set_religious_school = {
			group = raheni
			school = unbroken_claw_school
		}
	}
	
	#Fangaulan - Administrative Candidate
	option = {
		trigger = { NOT = { has_ruler_flag = bhuvauri_fangaulan_ruler } }
		name = flavor_bhuvauri.2.b
		if = {
			limit = { republic_with_female_rulers_trigger = yes }
			define_ruler = {
				max_random_mil = 1
				max_random_adm = 3
				max_random_dip = 2
				change_adm = 2
				change_dip = 2
				change_mil = 1
				culture = taafi
				religion = fangaula_pantheon
				random_gender = yes # Allows random gender for republics
			}
		}
		else = {
			define_ruler = {
				max_random_mil = 1
				max_random_adm = 3
				max_random_dip = 2
				change_adm = 2
				change_dip = 2
				change_mil = 1
				culture = taafi
				religion = fangaula_pantheon
			}
		}
		hidden_effect = {
			clr_ruler_flag = bhuvauri_bulwari_ruler
			clr_ruler_flag = bhuvauri_dakinshi_ruler
			set_ruler_flag = bhuvauri_fangaulan_ruler
		}
		set_religious_school = {
			group = raheni
			school = orange_sash_school
		}
	}
	
	#Bulwari - Diplomatic Candidate
	option = {
		trigger = { NOT = { has_ruler_flag = bhuvauri_bulwari_ruler } }
		name = flavor_bhuvauri.2.c
		if = {
			limit = { republic_with_female_rulers_trigger = yes }
			define_ruler = {
				max_random_mil = 2
				max_random_adm = 1
				max_random_dip = 3
				change_adm = 1
				change_dip = 2
				change_mil = 2
				culture = surani
				religion = high_philosophy
				dynasty = "szel-Arcas"
				random_gender = yes # Allows random gender for republics
			}
		}
		else = {
			define_ruler = {
				max_random_mil = 2
				max_random_adm = 1
				max_random_dip = 3
				change_adm = 1
				change_dip = 2
				change_mil = 2
				culture = surani
				religion = high_philosophy
				dynasty = "szel-Arcas"
			}
		}
		hidden_effect = {
			clr_ruler_flag = bhuvauri_bulwari_ruler
			clr_ruler_flag = bhuvauri_dakinshi_ruler
			set_ruler_flag = bhuvauri_fangaulan_ruler
		}
		set_religious_school = {
			group = raheni
			school = silk_turban_school
		}
	}
