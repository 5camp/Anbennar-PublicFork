#Gommoport
country_event = {
	id = aelantirspawnables.1
	title = aelantirspawnables.1.t
	desc = aelantirspawnables.1.d
	picture = CARIBBEAN_PIRATE_FORT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_global_flag = new_world_discovered
		any_province = {
			area = gommiochand_area
			is_empty = yes
			
			#has_discovered = ROOT
			
			has_empty_adjacent_province = yes
		}

		
		is_year = 1570
		
		# OR = {	#sort this out make it so player
			# if = {
				# limit = { 
					# capital_scope = { continent = north_america
				# }
			# }
		# }
	}
	
	mean_time_to_happen = {
		months = 600
		
		modifier = {
			factor = 0.1
			current_age = age_of_revolutions
		}
		modifier = {
			factor = 0.5
			exists = A79 #gnomish hierarchy
		}
		modifier = {
			factor = 0.5
			any_country = {
				capital_scope = { continent = europe }
				has_discovered = 2088
			}
		}
	}
	
	immediate = {
		
		hidden_effect = {
		
		random_province = {
			limit = {
				area = gommiochand_area
				is_empty = yes
				
				has_empty_adjacent_province = yes
				
			}
			
			save_event_target_as = spawn_province
		}
		
		random_province = {
			limit = {
				area = gommiochand_area
				is_empty = yes
				
				has_empty_adjacent_province = yes
				
				has_port = yes
			}
			
			save_event_target_as = spawn_province
		}		

		random_province = {
			limit = {
				area = gommiochand_area
				is_empty = yes
				
				has_empty_adjacent_province = yes
				
				province_id = 2088
			}
			
			save_event_target_as = spawn_province
		}
		
		}
	}
	
	option = {
		name = aelantirspawnables.1.a
		
		event_target:spawn_province = {
			create_colony = 1000
		}
		
		event_target:spawn_province = {
			cede_province = H30
		}
		
		hidden_effect = {
			H30 = {
				country_event = {
					id = anb_miscevents.7
					days = 30
				}
			}
		}
	}
	
	option = { # play as the pirates
		name = aelantirspawnables.1.b
		trigger = {
			ai = no
		}
		event_target:spawn_province = {
			create_colony = 1000
		}
		
		event_target:spawn_province = {
			cede_province = H30
		}

		switch_tag = H30
		
		hidden_effect = {
			H30 = {
				country_event = { 
					id = anb_miscevents.7
					days = 30
				}
			}
		}
		
	}
