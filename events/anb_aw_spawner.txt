#0 - spawn in province
country_event = {
    id = aw_spawner.0
    title = aw_spawner.0.t
    desc = aw_spawner.0.d
    picture = COMET_SIGHTED_eventPicture
    
    is_triggered_only = yes
    hidden = yes

    immediate = {
        random_owned_province = {
            limit = {
                is_city = yes
            }
            save_event_target_as = aw_spawn_target
        }
    }
    
    option = {
        goto = aw_spawn_target
        event_target:aw_spawn_target = {
            random_list = {
                1 = {
                    #Preventers
                    modifier = {
                        factor = 0
                        NOT = { development = 20 }
                    }

                    #Factors
                    modifier = {    #Terrain Type
                        factor = 2
                        has_terrain = city_terrain
                    }
                    modifier = {
                        factor = 1.5
                        development = 20
                    }
                    modifier = {
                        factor = 1.25
                        development = 30
                    }
                    modifier = {
                        factor = 2
                        owner = { NOT = { stability = 1 }}
                    }
                    province_event = { id = aw_thieves_guild.0 }
                }
                3 = {
                    #Preventers
                    modifier = {
                        factor = 0
                        has_terrain = city_terrain  #bandits cannot spawn in cities
                    }

                    #Factors
                    modifier = {    #Terrain Type
                        factor = 3
                        OR = {
                            has_terrain = forest
                            has_terrain = hills
                        }
                    }
                    modifier = {
                        factor = 2
                        devastation = 25
                    }
                    modifier = {
                        factor = 2
                        devastation = 50
                    }
                    modifier = {
                        factor = 2
                        devastation = 75
                    }
                    modifier = {
                        factor = 2
                        owner = { NOT = { stability = 1 }}
                    }
                    province_event = { id = aw_bandits.0 }
                }
                
            }
        }   
    }
}
