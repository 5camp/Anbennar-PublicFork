				estate = estate_burghers
				share = 15
			}
		}
		ai_chance = { factor = 100 }
	}
}

	trigger = {
		#primary_culture = cave_goblin
	}

	is_triggered_only = yes
	hidden = yes

	immediate = {
		if = {
			limit = {
				has_country_flag = cartel_level_autofire
				NOT = {
					has_country_modifier = cartel_control_level_1
					has_country_modifier = cartel_control_level_2
					has_country_modifier = cartel_control_level_3
					has_country_modifier = cartel_control_level_4
					has_country_modifier = cartel_control_level_5
				}
			}
			remove_country_flag = cartel_level_autofire
			country_event = {
				id = cartel_wars_events.3
				days = 0
			}
		}
		else_if = {
			limit = {
				has_country_flag = cartel_level_autofire
			}
			remove_country_flag = cartel_level_autofire
		}
		else_if = {
			limit = {
				has_country_modifier = cartel_control_level_1
			}
			country_event = {
				id = cartel_wars_events.4
				days = 0
			}
		}
		else_if = {
			limit = {
				has_country_modifier = cartel_control_level_2
			}
			country_event = {
				id = cartel_wars_events.5
				days = 0
			}
		}
		else_if = {
			limit = {
				has_country_modifier = cartel_control_level_3
			}
			country_event = {
				id = cartel_wars_events.6
				days = 0
			}
		}
		else_if = {
			limit = {
				has_country_modifier = cartel_control_level_4
			}
			country_event = {
				id = cartel_wars_events.7
				days = 0
			}
		}
		else_if = {
			limit = {
				has_country_modifier = cartel_control_level_5
			}
			country_event = {
				id = cartel_wars_events.8
				days = 0
			}
		}
	}

	option = {
		name = cartel_wars_events.2.a

		ai_chance = { factor = 100 }
	}
}
