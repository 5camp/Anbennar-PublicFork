country_event = {
	id = monstrous.1
	title = monstrous.1.t
	desc = monstrous.1.d
	picture = REFORM_eventPicture
	
	trigger = {
		has_country_modifier = monstrous_nation
		has_country_flag = removing_monstrous
		NOT = { has_country_flag = half_monstrous }
		NOT = { check_variable = { monstrousPoints = 50 } }
	}
	
	mean_time_to_happen = {
		months = 6
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = half_monstrous
		}
	}
	
	option = {
		name = monstrous.1.a
		ai_chance = { factor = 100 }
		custom_tooltip = half_monstrous_tooltip
		hidden_effect = {
			every_country = {
				limit = { has_opinion_modifier = { who = ROOT modifier = root_monstrous } }
				remove_opinion = { who = ROOT modifier = root_monstrous }
				add_opinion = { who = ROOT modifier = half_monstrous }
			}
			every_country = {
				limit = { reverse_has_opinion_modifier = { who = ROOT modifier = root_monstrous } }
				reverse_remove_opinion = { who = ROOT modifier = root_monstrous }
				reverse_add_opinion = { who = ROOT modifier = half_monstrous }
			}
		}
	}
}

country_event = {
	id = monstrous.4
	title = monstrous.4.t
	desc = monstrous.4.d
	picture = REFORM_eventPicture
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	trigger = {
		has_country_modifier = monstrous_nation
		has_country_flag = removing_monstrous
		any_neighbor_country = {
			NOT = { has_country_modifier = monstrous_nation }
			any_owned_province = { is_claim = ROOT }
			NOT = { war_with = ROOT }
		}
	}
	
	immediate = {
		random_neighbor_country = {
			limit = { 
				any_owned_province = { is_claim = ROOT }
				NOT = { has_country_modifier = monstrous_nation }
				NOT = { war_with = ROOT }
			}
			save_event_target_as = country
			random_owned_province = {
				limit = { is_claim = ROOT }
				save_event_target_as = province
			}
		}
	}
	#Remove Claim
	option = {
		name = monstrous.4.a
		ai_chance = { factor = 1 }
		decrease_monstrous_1 = yes
		add_adm_power = -50
		event_target:province = {
			remove_claim = ROOT
		}
	}
	#Keep claim
	option = {
		name = monstrous.4.b
		ai_chance = { 
			factor = 100
			
			modifier = {
				factor = 0
				want_to_decrease_monstrous = yes
			}
		}
		add_adm_power = 100
		increase_monstrous_1 = yes
	}
}
