#X00 - hags tiers up with people moving to her province and some going missing
province_event = {
    id = aw_wyvern.X00
    title = aw_wyvern.X00.t
    desc = aw_wyvern.X00.d
    picture = COMET_SIGHTED_eventPicture
    goto = ROOT
    
    hidden = no
    is_triggered_only = yes
    
    trigger = {
        owner = {
            any_province = {
                region = ROOT
                OR = {
                    has_province_modifier = aw_hags_1 #your region has a hag
                    has_province_modifier = aw_hags_2
                }
            }
        }
#        region_for_scope_province = {
#            OR = {
#                has_province_modifier = aw_hags_1 #your region has a hag
#                has_province_modifier = aw_hags_2
#            }
#        }
    }
    immediate = {
        hidden_effect = {
            random_province = {
                limit = {
                    region = ROOT
                    owner = ROOT
                    OR = {
                        has_province_modifier = aw_hags_1 #your region has a hag
                        has_province_modifier = aw_hags_2
                    }
                }
                save_event_target_as = nested_wyvern_origin_province
            }
        }
    }
    option = {
        name = aw_wyvern.X00.a
        ai_chance = { factor = 100 }

        add_local_autonomy = 10
        random_list = {
            33 = {
                add_base_tax = -1
                add_base_production = -1
            }
            33 = {
                add_base_production = -1
                add_base_manpower = -1
            }
            33 = {
                add_base_tax = -1
                add_base_manpower = -1
            }
        }
        event_target:nested_wyvern_origin_country = {
            random_list = {
            33 = {
                add_base_tax = 1
            }
            33 = {
                add_base_production = 1
            }
            33 = {
                add_base_manpower = 1
            }
        }
            if = {
                limit = { 
                    has_province_modifier = aw_hags_1
                }
                remove_province_modifier = aw_hags_1
                add_permanent_province_modifier  = {
                    name = aw_hags_2
                    duration = -1  
                }
            }
            else_if = {
                limit = { 
                    has_province_modifier = aw_hags_2
                }
                remove_province_modifier = aw_hags_2
                add_permanent_province_modifier  = {
                    name = aw_hags_3
                    duration = -1  
                }
            }
        }
    }
}

#X - mages bargain
country_event = {
    id = aw_hags.X
    title = aw_hags.X.t
    desc = aw_hags.X.d
    picture = COMET_SIGHTED_eventPicture
    
    hidden = no
    is_triggered_only = yes
    
    trigger = {
        any_owned_province = {
            OR = {
                has_province_modifier = aw_hags_1
                has_province_modifier = aw_hags_2
                has_province_modifier = aw_hags_3
            }
        }
        NOT = { has_country_modifier = aw_hags_mage_estate }
    }
    
    option = {
 
        name = aw_hags.X.a
        ai_chance = { factor = 100 }

        if = {
            limit = {
                has_province_modifier = aw_hags_1
            }
            add_country_modifier = {
                name = aw_hags_mage_estate
                duration = 730
            }
        }
        else_if = {
            limit = {
                has_province_modifier = aw_hags_2
            }
            add_country_modifier = {
                name = aw_hags_mage_estate
                duration = 1825
            }
        }
        else_if = {
            limit = {
                has_province_modifier = aw_hags_3
            }
            add_country_modifier = {
                name = aw_hags_mage_estate
                duration = 3650
            }
        }
    }
}
