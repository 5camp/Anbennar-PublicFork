
namespace = flavor_khozrugan

#Chief Artug Ostokhan
country_event = {
	id = flavor_khozrugan.1
	title = flavor_khozrugan.1.t
	desc = flavor_khozrugan.1.d
	picture = FORT_eventPicture
	
	fire_only_once = yes

	trigger = {
		tag = Z51
		has_regency = no
		has_consort_regency = no
		is_lesser_in_union = no
		is_year = 1500
		NOT = { is_year = 1600 }
	}
	
	#Give up the throne
	option = {
		name = flavor_khozrugan.1.a
		ai_chance = {
			factor = 80
		}
		add_dip_power = 100
		define_ruler = {
			name = "Artug"
			dynasty = "Ostokhan"
			ADM = 5
			DIP = 5
			MIL = 6
			fixed = yes
			age = 21
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = bold_fighter_personality }
		}
		define_heir = {
			name = "Ornog"
			dynasty = "Ostokhan"
			culture = green_orc
			adm = 4
			dip = 4
			mil = 5
			age = 0
		}
		add_estate_loyalty = {
			estate = estate_tribes
			loyalty = 10
		}
	}
	
	#Fight! (lose)
	option = {
		name = flavor_khozrugan.1.b
		trigger = {
			NOT = { mil = 6 }
		}
		ai_chance = {
			factor = 20 
		}
		custom_tooltip = lets_fight_tooltip
		hidden_effect = {
			country_event = { id = flavor_khozrugan.3 days = 1 }
		}
	}
	
	#Fight! (win)
	option = {
		name = flavor_khozrugan.1.c
		trigger = {
			mil = 6
		}
		ai_chance = {
			factor = 20 
		}
		custom_tooltip = lets_fight_tooltip
		hidden_effect = {
			country_event = { id = flavor_khozrugan.3 days = 1 }
		}
	}
}
