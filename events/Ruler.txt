#Ruler Dies - for over age 100
country_event = {
	id = anb_ruler.4
	title = anb_ruler.4.t
	desc = anb_ruler.4.d
	picture = KING_SICK_IN_BED_eventPicture

	is_triggered_only = yes

	trigger = {
		has_ruler_flag = end_life
		ruler_age = 100
		is_lesser_in_union = no
	}

	option = {
		name = anb_ruler.4.a

		trigger = {
			OR = {
				adm = 2
				dip = 2
				mil = 2
			}
		}

		if = {
			limit = {
				adm = 6
				dip = 6
				mil = 6
			}
			custom_tooltip = anb_ruler.4.tt.1
		}
		else_if = {
			limit = {
				adm = 6
			}
			custom_tooltip = anb_ruler.4.tt.2
		}
		else_if = {
			limit = {
				mil = 6
			}
			custom_tooltip = anb_ruler.4.tt.3
		}
		else_if = {
			limit = {
				dip = 6
			}
			custom_tooltip = anb_ruler.4.tt.4
		}
		else_if = {
			limit = {
				ruler_age = 375
			}
			custom_tooltip = anb_ruler.4.tt.5
		}
		else = {
			custom_tooltip = anb_ruler.4.tt.6
		}
	}
	option = {
		name = anb_ruler.4.b
		trigger = {
			NOT = {
				OR = {
					adm = 2
					dip = 2
					mil = 2
				}
			}
		}
		custom_tooltip = anb_ruler.4.tt.7
	}

	after = {
		hidden_effect = { kill_ruler = yes }
		log = "[Root.GetName]'s ruler' died. Over 100 years old."
	}
}
