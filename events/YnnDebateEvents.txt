country_event = { #subject choose side on debate 1
	id = ynn_debate.12
	title = ynn_debate.12.t
	desc = ynn_debate.12.d
	picture = ENCOMIENDA_eventPicture

	is_triggered_only = yes
	trigger = {
		always = yes
	}

	option = {
		name = ynn_debate.12.a
		trigger = {
			OR = {
				#ai = no
				
				NOT = { has_country_flag = force_yrw_1b }
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 10
				has_country_flag = yrw_1a
			}
			modifier = {
				factor = 20
				OR = {
					tag = G39
					tag = G34
				}
			}
		}
		clr_country_flag = yrw_1b
		clr_country_flag = force_yrw_1a
		set_country_flag = yrw_1a
		add_prestige = 5
	}

	option = {
		name = ynn_debate.12.b
		trigger = {
			OR = {
				#ai = no
				
				NOT = { has_country_flag = force_yrw_1a }
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 10
				has_country_flag = yrw_1b
			}
			modifier = {
				factor = 20
				OR = {
					tag = G38
					tag = G30
					tag = G26
				}
			}
		}
		clr_country_flag = yrw_1a
		clr_country_flag = force_yrw_1b
		set_country_flag = yrw_1b
		add_prestige = 5
	}
	
	after = {
		if = {
			limit = {
				overlord = {	#When there's only one undecided iosahar left
					NOT = {
						calc_true_if = {
							any_subject_country = {
								is_subject_of_type = ynnic_iosahar
								NOT = {
									has_country_flag = yrw_1a
									has_country_flag = yrw_1b
								}
							}
						amount = 2
						}
						AND = {
                            any_subject_country = {
                                has_country_flag = yrw_1a
                            }
                            any_subject_country = {
                                has_country_flag = yrw_1b
                            }
                        }
					}
				}
			}
			if = {
				limit = {
					has_country_flag = yrw_1a
				}
				overlord = {
					random_subject_country = {
						limit = {
							is_subject_of_type = ynnic_iosahar
							NOT = {
								has_country_flag = yrw_1a
								has_country_flag = yrw_1b
							}
						}
						set_country_flag = force_yrw_1b
					}
				}
			}
			else_if = {
				limit = {
					has_country_flag = yrw_1b
				}
				overlord = {
					random_subject_country = {
						limit = {
							is_subject_of_type = ynnic_iosahar
							NOT = {
								has_country_flag = yrw_1a
								has_country_flag = yrw_1b
							}
						}
						set_country_flag = force_yrw_1a
					}
				}
			}	
		}
		
		overlord = {
			random_subject_country = {
				limit = {
					is_subject_of_type = ynnic_iosahar
					NOT = {
						has_country_flag = yrw_1a
						has_country_flag = yrw_1b
					}
				}
				country_event = { id = ynn_debate.12 }
			}
		}
	}
}
