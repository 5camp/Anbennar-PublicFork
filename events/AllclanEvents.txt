country_event = { #end of crisis
	id = allclan.910
	title = allclan.910.t
	desc = allclan.910.d
	picture = TRADEGOODS_eventPicture
	
	major = no

	trigger = {
		has_reform = nezkaru_council_reform
	}

	is_triggered_only = yes
	
	option = { #resolve the crisis
		name = allclan.910.a
		ai_chance = { factor = 100 }
		#allclan
		trigger = {
			OR = {
				stability = 3
				republican_tradition = 75
				AND = {
					NOT = {
						faction_influence = {
							faction = allclan_reformists_faction
							influence = 40
						}
					}
					AND = {
						faction_influence = {
							faction = allclan_cartels_faction
							influence = 25
						}
						faction_influence = {
							faction = allclan_technocrats_faction
							influence = 25
						}
					}
				}
				AND = {
					NOT = {
						faction_influence = {
							faction = allclan_cartels_faction
							influence = 40
						}
					}
					AND = {
						faction_influence = {
							faction = allclan_reformists_faction
							influence = 25
						}
						faction_influence = {
							faction = allclan_technocrats_faction
							influence = 25
						}
					}
				}
				AND = {
					NOT = {
						faction_influence = {
							faction = allclan_technocrats_faction
							influence = 40
						}
					}
					AND = {
						faction_influence = {
							faction = allclan_cartels_faction
							influence = 25
						}
						faction_influence = {
							faction = allclan_reformists_faction
							influence = 25
						}
					}
				}
			}
		}
		add_republican_tradition = 10
		hidden_effect = {
			set_country_flag = allclan_resolved_council_disturbance
		}
	}

	option = { #step down to resolve issues
		name = allclan.910.b
		ai_chance = { factor = 2 }
		#allclan
		kill_ruler = yes
		random = { 
			chance = 50 
		
			add_stability = 1
			add_republican_tradition = 5
		}
		if = {
			limit = {
				has_ruler_modifier = allclan_technocrat_ruler
			}
			add_faction_influence = {
				faction = allclan_technocrats
				influence = -10
			}
		}
		if = {
			limit = {
				has_ruler_modifier = allclan_cartel_ruler
			}
			add_faction_influence = {
				faction = allclan_cartels
				influence = -10
			}
		}
		if = {
			limit = {
				has_ruler_modifier = allclan_reformist_ruler
			}
			add_faction_influence = {
				faction = allclan_reformists
				influence = -10
			}
		}
	}

	option = { #work on issues
		name = allclan.910.c
		ai_chance = { factor = 1 }
		#allclan
		random = { 
			chance = 33 
	
			add_republican_tradition = 5
		}
	}
}
