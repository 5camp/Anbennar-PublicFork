country_event = {
	id = fey_court.1
	title = fey_court.1.t
	desc = fey_court.1.d
	picture = FEAST_eventPicture
	
	is_triggered_only = yes

	trigger = {
		religion = fey_court	
	}

	option = {
		name = fey_court.1.a
		add_yearly_manpower = -0.5
		fey_court_slightly_improve_seelie_court_relations = true
		fey_court_slightly_hurt_unseelie_court_relations = true
	}
	option = {
		name = fey_court.1.b
		dip_power_cost = 50
		fey_court_moderately_improve_seelie_court_relations = true
		fey_court_slightly_hurt_unseelie_court_relations = true
	}
	option = {
		name = fey_court.1.c
		capital_scope = {
			add_devastation = 25
		}
		fey_court_greatly_improve_seelie_court_relations = true
		fey_court_slightly_hurt_unseelie_court_relations = true
	}
}
