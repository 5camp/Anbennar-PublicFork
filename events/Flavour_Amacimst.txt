#The Two Castles
country_event = {
	id = flavour_amacimst.6
	title = flavour_amacimst.6.t
	desc = flavour_amacimst.6.d
	picture = CASTLE_eventPicture

	is_triggered_only = yes

	trigger = {
		tag = G26
	}

	option = {	#Reinstate Amaceped
		name = flavour_amacimst.6.a
		ai_chance = {
			factor = 1.5
		}
		if = {
			limit = {
				1168 = {
					NOT = {
						owned_by = G30
					}
				}
			}
			1168 = {
				cede_province = G30
			}
		}
		G30 = {
			override_country_name = AMACEPED
			add_opinion = {
				who = ROOT
				modifier = historical_friend
			}
		}
		add_opinion = {
			who = G30
			modifier = historical_friend
		}
	}

	option = {	#Keep The Castle
		name = flavour_amacimst.6.b
		ai_chance = {
			factor = 1
			modifier = {
				factor = 2
				ruler_has_personality = greedy_personality
			}
		}
		if = {
			limit = {
				1168 = {
					NOT = {
						owned_by = ROOT
					}
				}
			}
			1168 = {
				cede_province = ROOT
			}
		}
		1168 = {
			add_province_modifier = {
				name = G26_amethyst_order_fortress
				duration = -1
			}
		}
	}
}
