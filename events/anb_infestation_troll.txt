namespace = infestation_troll

# for testing, use event infestation_troll.0 to force spawn in one of your provinces

# notes on trolls:
# troll infestations have three sizes, 1, 2, 3 with three being the largest
# trolls can be civilized as racial minority
# trolls can rebel at size 2 or 3, with demands to civilize

# possible flags and province modifiers
# infestation_present - province flag, universal
# infestation_troll_1, infestation_troll_2, infestation_troll_3 for sizes of trolls
# infestation_troll_stronghold - modifier which affects the fort maintenance/defensiveness.
# infestation_troll_rebels - country flag so that we can clean up after a rebellion

# spawn troll infestation in this province and start event loop
province_event = {
    id = infestation_troll.0
    title = infestation_troll.0.t
    desc = infestation_troll.0.d
    picture = COMET_SIGHTED_eventPicture
    goto = ROOT
    
    hidden = yes
    is_triggered_only = yes
    
    immediate = {
        hidden_effect = {
            set_province_flag = infestation_present # important; or event loop will exit
        }
    }
    
    option = {
        ai_chance = { factor = 100 }
        add_permanent_province_modifier  = {
            name = infestation_troll_1
            duration = -1  
            desc = infestation_troll_1_tooltip
        }
        province_event = { id = infestation_troll.100 days = 1 } # spawn response event
    }
    after = {
        # first event in half a year, ish; start event loop
        #province_event = { id = infestation_troll.1 days = 100 random = 100 }
        set_province_flag = infestation_pulse_flag #we don't want to start pulsing yet
        add_province_triggered_modifier = infestation_troll_monthly_pulse # we don't want this to start pulsing yet
    }
}

# random event dispatcher, on a loop until infestation_present is removed
province_event = {
    id = infestation_troll.1
    title = infestation_troll.1.t
    desc = infestation_troll.1.d
    picture = COMET_SIGHTED_eventPicture
    goto = ROOT
    
    hidden = yes
    is_triggered_only = yes
    

    trigger = {
        has_province_flag = infestation_present
        OR = { 
            has_province_modifier = infestation_troll_1
            has_province_modifier = infestation_troll_2
            has_province_modifier = infestation_troll_3
        }
    }
	
	option = {
		hidden_effect = {
			random_list = {
				1 = { # vanish
					trigger = { has_province_modifier = infestation_troll_1 }
					province_event = { id = infestation_troll.101 }
				}
				1 = {
					trigger = {
						NOT = { has_province_modifier = infestation_troll_stronghold } # stops migration
						owner = { num_of_cities = 2 } # you have something to migrate to
						any_neighbor_province = { 
							owned_by = ROOT
							NOT = { has_province_flag = infestation_present }
						}
					}
					province_event = { id = infestation_troll.110 }
				}
				1 = { # migrate across border
					trigger = {
						NOT = { has_province_modifier = infestation_troll_stronghold } # stops migration
						any_neighbor_province = { 
							NOT = { owned_by = ROOT }
							NOT = { has_province_flag = infestation_present }
						}
					}
					province_event = { id = infestation_troll.111 }
				}
				1 = { # banish (migrate to random neighbour)
					modifier = {
						factor = 3 # more likely if generous quest rewards/adverturer nation
						OR = {
							owner = { has_estate_privilege = estate_adventurers_generous_quest_rewards }
							owner = { has_reform = adventurer }
						}
					}
					modifier = {
						factor = 5 # if being expelled through racial menu
						owner = { has_country_modifier = racial_pop_troll_expulsion }
					}
					province_event = { id = infestation_troll.115 }
				}
				1 = { # Spreads internally
					trigger = { 
						OR = { 
							has_province_modifier = infestation_troll_2 
							has_province_modifier = infestation_troll_3
						}
					}
					province_event = { id = infestation_troll.120 }
				}
				1 = { # Spreads across borders
					trigger = { 
						OR = { 
							has_province_modifier = infestation_troll_2 
							has_province_modifier = infestation_troll_3
						}
					}
					province_event = { id = infestation_troll.121 }
				}
				2 = { # grows
					trigger = { 
						OR = { 
							has_province_modifier = infestation_troll_1 
							has_province_modifier = infestation_troll_2
						}
					}
					province_event = { id = infestation_troll.130 }
				}
				1 = { # shrinks
					trigger = { 
						OR = { 
							has_province_modifier = infestation_troll_2 
							has_province_modifier = infestation_troll_3
						}
					}
					province_event = { id = infestation_troll.131 }
				}
				3 = { # devastation
					province_event = { id = infestation_troll.140 }
				}
				1 = { # provoke rebellion
					trigger = { 
						OR = { 
							has_province_modifier = infestation_troll_2 
							has_province_modifier = infestation_troll_3
						}
					}
					modifier = {
						factor = 3 # more likely if generous quest rewards/adverturer nation
						OR = {
							owner = { has_estate_privilege = estate_adventurers_generous_quest_rewards }
							owner = { has_reform = adventurer }
						}
					}
					province_event = { id = infestation_troll.146 }
				}
				1 = { # infestation killed -> shrinks/vanishes
					modifier = {
						factor = 3 # more likely if generous quest rewards/adverturer nation
						OR = {
							owner = { has_estate_privilege = estate_adventurers_generous_quest_rewards }
							owner = { has_reform = adventurer }
						}
					}
					modifier = {
						factor = 5 # if being purged through racial menu
						owner = { has_country_modifier = racial_pop_troll_purge }
					}
					province_event = { id = infestation_troll.150 }
				}
				1 = { # civilizes
					trigger = {
						owner = { medium_tolerance_troll_race_trigger = yes } # Need to already like them somewhat
					}
					modifier = {
						factor = 5 # trolls infest a troll nation! Let's incorporate them!
						OR = {
							owner = { culture_group = troll } # country is in culture group
							owner = { has_troll_accepted_culture = yes }
							culture_group = troll # province is in culture group
						}
					}
					modifier = {
						factor = 5 # trolls already in province, and are an integrated minority
						OR = {
							has_troll_minority_trigger = yes
							has_troll_majority_trigger = yes
						}
						owner = { high_tolerance_troll_race_trigger = yes }
					}
					province_event = { id = infestation_troll.160 }
				}
			}
		}
	}
    
    #after = {
    #    # loop this random event dispatcher
    #    # about a year between events
    #    #province_event = { id = infestation_troll.1 days = 300 random = 100 }
    #}
}
