	trigger = {
		religion = chahinanito
		NOT = {
			has_country_flag = effelai_worship_all_holy_sites_visited
		}
	}

		set_country_flag = effelai_worship_brain_visited

			factor = 100
		}

			name = effelai_brain_visited

		}
	}
}
