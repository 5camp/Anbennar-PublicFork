		tag = I98

			AND = {
				NOT = { mission_completed = I98_rabid_constructions }

					flag = I98_frenzy_flag

				}

				mission_completed = I98_rabid_constructions

					flag = I98_frenzy_flag

				}

				mission_completed = I98_delirium_of_the_hammer
				had_country_flag = {
					flag = I98_frenzy_flag
					days = 16425
				}
