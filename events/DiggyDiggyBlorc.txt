country_event = {
	id = blorc.50
	title = blorc.50.t
	desc = blorc.50.d
	picture = CIVIL_WAR_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		always = yes
	}
	
	immediate = {
		hidden_effect = {
			define_hold_to_duel = yes
		}
	}
	
	after = {
	}
	
	option = {
        name = blorc.50.a
        ai_chance = { factor = 5 }
		trigger = { has_saved_event_target = duel1 }
		goto = duel1
		event_target:duel1 = { save_event_target_as = blorc_duel_target }
		country_event = { id = blorc.51 }
		add_years_of_income = -0.1
		add_mil_power = -50
	}
	
	option = {
        name = blorc.50.b
        ai_chance = { factor = 5 }
		trigger = { has_saved_event_target = duel2 }
		goto = duel2
		event_target:duel2 = { save_event_target_as = blorc_duel_target }
		country_event = { id = blorc.51 }
		add_years_of_income = -0.1
		add_mil_power = -50
	}
	
	option = {
        name = blorc.50.c
        ai_chance = { factor = 5 }
		trigger = { has_saved_event_target = duel3 }
		goto = duel3
		event_target:duel3 = { save_event_target_as = blorc_duel_target }
		country_event = { id = blorc.51 }
		add_years_of_income = -0.1
		add_mil_power = -50
	}
	
	option = {
        name = blorc.50.e
        ai_chance = { factor = 5 }
		trigger = { has_saved_event_target = duel4 }
		goto = duel4
		event_target:duel4 = { save_event_target_as = blorc_duel_target }
		country_event = { id = blorc.51 }
		add_years_of_income = -0.1
		add_mil_power = -50
	}
	
	#Go Back
	option = {
        name = blorc.50.f
        ai_chance = { factor = 0 }
		hidden_effect = {
			every_owned_province = {
				limit = { has_province_flag = duel_@ROOT }
				clr_province_flag = duel_@ROOT
				clr_province_flag = option_duel_1_@ROOT
				clr_province_flag = option_duel_2_@ROOT
				clr_province_flag = option_duel_3_@ROOT
				clr_province_flag = option_duel_4_@ROOT
			}
			clr_country_flag = open_duel_hold_menu
		}
	}
}

	desc = blorc.52.d
	picture = CIVIL_WAR_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		always = yes
	}
	
	option = {
        name = blorc.52.a
        ai_chance = { factor = 400 }
		custom_tooltip = blorc_duel_win_tooltip
		add_ruler_modifier = {
			name = blorc_recovering_from_ozarm
			duration = 1825
		}
		hidden_effect = {
			event_target:blorc_duel_target = {
				add_colonysize = 1000
				change_culture = ROOT
				change_religion = ROOT
				remove_province_modifier = infested_hold
				add_permanent_province_modifier = {
					name = ruined_hold
					duration = -1
				}
			}
		}
	}
	after = {
		every_owned_province = {
			limit = { has_province_flag = duel_@ROOT }
			clr_province_flag = duel_@ROOT
			clr_province_flag = option_duel_1_@ROOT
			clr_province_flag = option_duel_2_@ROOT
			clr_province_flag = option_duel_3_@ROOT
			clr_province_flag = option_duel_4_@ROOT
		}
		clr_country_flag = open_duel_hold_menu
	}
}
