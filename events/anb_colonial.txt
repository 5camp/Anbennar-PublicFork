# Colonial Migration
province_event = {
	id = anb_colonial.111
	title = "EVTNAME6401"
	desc = "EVTDESC6401"
	picture = COLONIZATION_eventPicture

	is_triggered_only = yes

	trigger = {
		OR = {
			region = west_castanor_region
			region = south_castanor_region
			region = inner_castanor_region
		}
		#colonysize = 300
		NOT = { colonysize = 500 }
		owner = {
			any_owned_province = {
				is_colony = yes
				colonysize = 500
				NOT = { province_id = ROOT }
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					region = west_castanor_region
				}
				export_to_variable = {
					variable_name = escanni_cultureVar
					value = { culture = adenner }
				}
			}
			if = {
				limit = {
					region = inner_castanor_region
				}
				export_to_variable = {
					variable_name = escanni_cultureVar
					value = castellyrian
				}
			}
			if = {
				limit = {
					region = south_castanor_region
				}
				export_to_variable = {
					variable_name = escanni_cultureVar
					value = { culture = marcher }
				}
			}
		}
	}

	option = {
		name = "EVTOPTA6401"
		add_colonysize = -200
		owner = {
			random_owned_province = {
				limit = {
					is_colony = yes
					colonysize = 500
					NOT = { province_id = ROOT }
				}
				add_colonysize = 200
			}
		}
	}
}

			add_faction_influence = {	#will need to make this into an effect later

			}
		}
	}
	
	#It's not safe enough, we need more fighters!
	option = {
		name = anb_colonial.3.b
		owner = {
			add_faction_influence = {	#will need to make this into an effect later

				influence = 10
			}
			add_prestige = 5
			add_yearly_manpower = 0.5
			add_years_of_income = -0.5
		}
	}
	
}
