namespace = infestation_werewolf

# for testing, use event infestation_werewolf.0 to force spawn in one of your provinces

# notes on werewolfs:
# werewolf infestations have three sizes, 1, 2, 3 with three being the largest
# werewolfs can be civilized as racial minority
# werewolfs can rebel at size 2 or 3, with demands to civilize

# possible flags and province modifiers
# infestation_present - province flag, universal
# infestation_werewolf_1, infestation_werewolf_2, infestation_werewolf_3 for sizes of werewolfs
# infestation_werewolf_stronghold - modifier which affects the fort maintenance/defensiveness.
# infestation_werewolf_rebels - country flag so that we can clean up after a rebellion

# spawn werewolf infestation in this province and start event loop
province_event = {
    id = infestation_werewolf.0
    title = infestation_werewolf.0.t
    desc = infestation_werewolf.0.d
    picture = COMET_SIGHTED_eventPicture
    goto = ROOT
    
    hidden = yes
    is_triggered_only = yes
    
    immediate = {
        hidden_effect = {
            set_province_flag = infestation_present # important; or event loop will exit
        }
    }
    
    option = {
        ai_chance = { factor = 100 }
        add_permanent_province_modifier  = {
            name = infestation_werewolf_1
            duration = -1  
            desc = infestation_werewolf_1_tooltip
        }
        province_event = { id = infestation_werewolf.100 days = 1 } # spawn response event
    }
    after = {
        # first event in half a year, ish; start event loop
        #province_event = { id = infestation_werewolf.1 days = 100 random = 100 }
        set_province_flag = infestation_pulse_flag #we don't want to start pulsing yet
        add_province_triggered_modifier = infestation_werewolf_monthly_pulse # we don't want this to start pulsing yet
    }
}

# random event dispatcher, on a loop until infestation_present is removed
province_event = {
    id = infestation_werewolf.1
    title = infestation_werewolf.1.t
    desc = infestation_werewolf.1.d
    picture = COMET_SIGHTED_eventPicture
    goto = ROOT
    
    hidden = yes
    is_triggered_only = yes
    

    trigger = {
        has_province_flag = infestation_present
        OR = { 
            has_province_modifier = infestation_werewolf_1
            has_province_modifier = infestation_werewolf_2
            has_province_modifier = infestation_werewolf_3
        }
    }
	
	option = {
		hidden_effect = {
			random_list = {
				1 = { # vanish
					trigger = { has_province_modifier = infestation_werewolf_1 }
					province_event = { id = infestation_werewolf.101 }
				}
				2 = {
					trigger = {
						owner = { num_of_cities = 2 } # you have something to migrate to
						any_neighbor_province = { 
							owned_by = ROOT
							NOT = { has_province_flag = infestation_present }
						}
					}
					province_event = { id = infestation_werewolf.110 }
				}
				1 = { # migrate across border
					trigger = {
						NOT = { has_province_modifier = infestation_werewolf_stronghold } # stops migration
						any_neighbor_province = { 
							NOT = { owned_by = ROOT }
							NOT = { has_province_flag = infestation_present }
						}
					}
					province_event = { id = infestation_werewolf.111 }
				}
				1 = { # banish (migrate to random neighbour)
					modifier = {
						factor = 3 # more likely if generous quest rewards/adverturer nation
						OR = {
							owner = { has_estate_privilege = estate_adventurers_generous_quest_rewards }
							owner = { has_reform = adventurer }
						}
					}
					modifier = {
						factor = 5 # if being expelled through racial menu
						owner = { has_country_modifier = racial_pop_werewolf_expulsion }
					}
					province_event = { id = infestation_werewolf.115 }
				}
				2 = { # Spreads internally
					trigger = { 
						OR = { 
							has_province_modifier = infestation_werewolf_2 
							has_province_modifier = infestation_werewolf_3
						}
					}
					province_event = { id = infestation_werewolf.120 }
				}
				1 = { # Spreads across borders
					trigger = { 
						OR = { 
							has_province_modifier = infestation_werewolf_2 
							has_province_modifier = infestation_werewolf_3
						}
					}
					province_event = { id = infestation_werewolf.121 }
				}
				2 = { # grows
					trigger = { 
						OR = { 
							has_province_modifier = infestation_werewolf_1 
							has_province_modifier = infestation_werewolf_2
						}
					}
					province_event = { id = infestation_werewolf.130 }
				}
				1 = { # shrinks
					trigger = { 
						OR = { 
							has_province_modifier = infestation_werewolf_2 
							has_province_modifier = infestation_werewolf_3
						}
					}
					province_event = { id = infestation_werewolf.131 }
				}
				3 = { # devastation
					province_event = { id = infestation_werewolf.140 }
				}
				1 = { # provoke rebellion
					trigger = { 
						OR = { 
							has_province_modifier = infestation_werewolf_2 
							has_province_modifier = infestation_werewolf_3
						}
					}
					modifier = {
						factor = 3 # more likely if generous quest rewards/adverturer nation
						OR = {
							owner = { has_estate_privilege = estate_adventurers_generous_quest_rewards }
							owner = { has_reform = adventurer }
						}
					}
					province_event = { id = infestation_werewolf.146 }
				}
				1 = { # infestation killed -> shrinks/vanishes
					modifier = {
						factor = 3 # more likely if generous quest rewards/adverturer nation
						OR = {
							owner = { has_estate_privilege = estate_adventurers_generous_quest_rewards }
							owner = { has_reform = adventurer }
						}
					}
					province_event = { id = infestation_werewolf.150 }
				}
			}
		}
	}
    
    #after = {
    #    # loop this random event dispatcher
    #    # about a year between events
    #    #province_event = { id = infestation_werewolf.1 days = 300 random = 100 }
    #}
}

# infestation migrates to adjacent province internally event
province_event = {
    id = infestation_werewolf.110
    title = infestation_werewolf.110.t
    desc = infestation_werewolf.110.d
    picture = COMET_SIGHTED_eventPicture
    goto = ROOT
    
    hidden = yes
    is_triggered_only = yes
    
    trigger = {
        owner = { num_of_cities = 2 } # you have something to migrate to
        any_neighbor_province = { 
            owned_by = ROOT
            NOT = { has_province_flag = infestation_present }
            has_influencing_fort = no # forts prevent infestations from spawning if active
        }
    }
    
    immediate = {
        hidden_effect = {
            random_neighbor_province = {
                limit = {
                    owned_by = ROOT
                    NOT = { has_province_flag = infestation_present }

                }
                save_event_target_as = infestation_migration_target
            }
            event_target:infestation_migration_target = {
                set_province_flag = infestation_present
            }
            remove_province_modifier = infestation_werewolf_stronghold # just in case
            hidden_effect = { remove_province_triggered_modifier = infestation_werewolf_monthly_pulse }
            cleanup_infestation = yes
        }
    }
    option = {
        name = infestation_werewolf.110.a
        ai_chance = { factor = 100 }
        goto = infestation_migration_target
        trigger = {
            has_province_modifier = infestation_werewolf_3
        }
        event_target:infestation_migration_target = {
            add_permanent_province_modifier  = {
                name = infestation_werewolf_3
                duration = -1  
                desc = infestation_werewolf_3_tooltip
            }
            hidden_effect = {
                add_province_triggered_modifier = infestation_werewolf_monthly_pulse
                set_province_flag = infestation_pulse_flag
            }
        }
        remove_province_modifier = infestation_werewolf_3
