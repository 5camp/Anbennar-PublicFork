#clear out the bilge
country_event = {
	id = anbenncost.2
	title = anbenncost.2.t
	desc = anbenncost.2.d
	picture = TRADEGOODS_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		always = yes
	}
	#renovate now
	option = {
		name = anbenncost.2.a
		ai_chance = { 
			factor = 100
			modifier = {
				factor = 0
				NOT = {
					treasury = 500
				}
			}
		}
		add_treasury = -500
		8 = {
			add_base_tax = 3
			add_base_production = 3
			add_base_manpower = 3
		}
	}
	#renovate in a decade
	option = {
		name = anbenncost.2.b
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				NOT = {
					treasury = 300
				}
			}
		}
		add_treasury = -300
		ROOT = {
			country_event = {
				id = anbenncost.3
				days = 3650
			}
		}
	}
	#renovate in 30 years
	option = {
		name = anbenncost.2.c
		ai_chance = {
			factor = 100
		}
		add_treasury = -100
		ROOT = {
			country_event = {
				id = anbenncost.3
				days = 10950
			}
		}
	}
}
