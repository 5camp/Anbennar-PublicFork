
namespace = cutdown

#Disaster - One country become too powerful
country_event = {
	id = cutdown.1
	title = cutdown.1.t
	desc = cutdown.1.d
	picture = FORT_eventPicture
	
	mean_time_to_happen = {
		years = 7
	}
	
	trigger = {
		has_country_modifier = lake_federation_member
		NOT = { has_global_flag = federation_crisis }
		NOT = { has_global_flag = federation_crisis_cutdown }
		NOT = { has_global_flag = lake_federation_disbanded }
		had_global_flag = {
			flag = last_federation_crisis
			days = 7300
		}
		is_lake_federation_leader = yes
		any_known_country = {
			has_country_modifier = lake_federation_member
			NOT = { has_country_modifier = federation_cutdown }			
			num_of_owned_provinces_with = {
				value = 13
				OR = {
					region = southern_isles_region
					region = northern_isles_region
					
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = { set_global_flag = federation_crisis_cutdown }
	}
	
	#We need to control them, ask them to reduce military expanditure
	option = {
		name = cutdown.1.a
		ai_chance = { factor = 50 }
		random_known_country = {
			limit = {
				has_country_modifier = lake_federation_member
				NOT = { has_country_modifier = federation_cutdown }
				num_of_owned_provinces_with = {
					value = 13
					OR = {
						region = southern_isles_region
						region = northern_isles_region
						
					}
				}
			}
			hidden_effect = { save_global_event_target_as = cutdown_country }
			hidden_effect = { set_country_flag = federation_red_leader }
			country_event = { id = cutdown.2 days = 10 }
		}
	}
}

#They ask us to cut down ou military spending
country_event = {
	id = cutdown.2
	title = cutdown.2.t
	desc = cutdown.2.d
	picture = FORT_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_country_modifier = lake_federation_member
	}
	
	#Accept
	option = {
		name = cutdown.2.a
		ai_chance = { factor = 5 }
		add_prestige = -5
		add_country_modifier = {
			name = federation_cutdown
			duration = 9125
		}
		FROM = { country_event = { id = cutdown.3 days = 10 } }
		lake_federation_gain_2_points = yes
	}
	
	#Refuse
	option = {
		name = cutdown.2.b
		ai_chance = { factor = 95 }
		FROM = { country_event = { id = cutdown.4 days = 10 } }
		lake_federation_lose_2_points = yes
		add_prestige = 5
		if = {
			limit = { alliance_with = FROM }
			add_historical_rival = FROM
		}
	}
}

	is_triggered_only = yes

	}

		add_country_modifier = {

			duration = 3650
		}
	}
	
	#Try to profit from it
	option = {
		name = cutdown.6.b
		ai_chance = { factor = 100 }
		add_mercantilism = 3
		random_owned_province = {
			limit = {
				OR = {
					trade_goods = copper
					trade_goods = iron
				}
			}
			add_base_production = 1
		}
	}
}

#The federation plunge into war
country_event = {
	id = cutdown.7
	title = cutdown.7.t
	desc = cutdown.7.d
	picture = FORT_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_country_modifier = lake_federation_member
	}
	
	#The war start
	option = {
		name = cutdown.7.a
		ai_chance = { factor = 100 }
		if = {
			limit = { has_country_modifier = federation_gearing_up }

		}
		else = {
			add_yearly_manpower = 2
		}
	}
}

#Enforce our rules
country_event = {
	id = cutdown.8
	title = cutdown.8.t
	desc = cutdown.8.d
	picture = FORT_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_country_modifier = lake_federation_member
	}
	
	#The war start
	option = {
		name = cutdown.8.a
		ai_chance = { factor = 100 }
		custom_tooltip = federation_plunge_into_war_tooltip
		if = {
			limit = { has_country_modifier = federation_gearing_up }
			add_yearly_manpower = 4
			custom_tooltip = federation_voluntary_tooltip
		}
		else = {
			add_yearly_manpower = 2
		}
		hidden_effect = {
			every_known_country = {
				limit = { has_country_flag = federation_red }
					create_guarantee = event_target:cutdown_country
			}
			form_coalition_against = event_target:cutdown_country
			every_known_country = {
				limit = { has_country_flag = federation_blue }
				form_coalition_against = event_target:cutdown_country
			}
			declare_war_with_cb = {
				who = event_target:cutdown_country
				casus_belli = cb_federation_war
			}
		}
	}
}

#End of the War
country_event = {
	id = cutdown.9
	title = cutdown.9.t
	desc = cutdown.9.d
	picture = FORT_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			has_country_flag = federation_red_leader
			has_country_flag = federation_blue_leader
		}
	}
	
	immediate = {
		hidden_effect = { clr_global_flag = federation_crisis }
		hidden_effect = { clr_global_flag = federation_crisis_cutdown }
		hidden_effect = { set_global_flag = last_federation_crisis }
	}
	
	#The Federation Triumph
	option = {
		name = cutdown.9.a
		ai_chance = { factor = 100 }
		trigger = { has_country_flag = federation_blue_leader }
		lake_federation_gain_10_points = yes
		every_known_country = {
			limit = { has_country_flag = federation_blue }
			lake_federation_gain_2_points = yes
		}
		every_known_country = {
			limit = { has_country_flag = federation_red }
			lake_federation_lose_2_points = yes
		}
		event_target:cutdown_country = {
			lake_federation_lose_10_points = yes
			add_country_modifier = {
				name = federation_cutdown
				duration = 18250
			}
		}
		hidden_effect = {
			every_known_country = {
				limit = { OR = { has_country_flag = federation_blue has_country_flag = federation_red } }
				clr_country_flag = federation_blue
				clr_country_flag = federation_red
				clr_country_flag = federation_red_leader
			}
			clr_country_flag = federation_blue
			clr_country_flag = federation_blue_leader
		}
	}
	
	#The Red Triumph
	option = {
		name = cutdown.9.b
		ai_chance = { factor = 100 }
		trigger = { has_country_flag = federation_red_leader }
		lake_federation_gain_10_points = yes
		every_known_country = {
			limit = { has_country_flag = federation_red }
			lake_federation_gain_2_points = yes
		}
		every_known_country = {
			limit = { has_country_flag = federation_blue }
			lake_federation_lose_2_points = yes
		}
		random_known_country = {
			limit = { has_country_flag = federation_blue_leader }
			lake_federation_lose_10_points = yes
		}
		hidden_effect = {
			every_known_country = {
				limit = { OR = { has_country_flag = federation_blue has_country_flag = federation_red } }
				clr_country_flag = federation_blue
				clr_country_flag = federation_red
				clr_country_flag = federation_blue_leader
			}
			clr_country_flag = federation_red
			clr_country_flag = federation_red_leader
		}
	}
}
