	id = migration_force_settle.1
	title = "theocracy.1.t"
	desc = "theocracy.1.desc"
	picture = RELIGION_eventPicture

	is_triggered_only = yes

	hidden = yes

	trigger = {
		OR = {
			has_reform = begin_settlement
			has_reform = native_settle_down_reform
			has_reform = greentide_horde
		}

		any_owned_province = {
			any_empty_neighbor_province = {
				is_empty = yes
				tribal_land_of = ROOT
			}
		}

		tribal_development = 10

		adm_power = 50
		dip_power = 50

	}
	
	# Settle a province
	option = {
		name = migration_force_settle.1.a
		ai_chance = {
			factor = 1

			modifier = {
				factor = 1.5
				adm_power = 100
				dip_power = 100
			}
			modifier = {
				factor = 2
				adm_power = 200
				dip_power = 200
			}

		}

		random = {
			chance = 50
			random_owned_province = {
				limit = {
					any_neighbor_province = {
						is_empty = yes
						tribal_land_of = ROOT
					}
				}
				random_empty_neighbor_province = {
					limit = {
						is_empty = yes
						tribal_land_of = ROOT
					}
					settle_province = ROOT
					if = {
						limit = { NOT = { culture = root religion = root } }
						change_culture = ROOT
						change_religion = ROOT
						add_devastation = 10
					}
				}
			}
			add_adm_power = -25
			add_dip_power = -25
		}

	}

	
	# Dont do anything
	option = {
		name = migration_force_settle.1.b	
		ai_chance = {
			factor = 1

			modifier = {
				factor = 1.5
				any_neighbor_country = {
					tech_difference = 3
				}
			}

			modifier = {
				factor = 1.25
				NOT = { stability = 0 }
			}

		}

		#Do fuck all
	}
	
	#Gauranteed
	option = {
		name = migration_force_settle.1.c	
		ai_chance = {
			factor = 1

			modifier = {
				factor = 10
				tribal_development = 20
			}

		}

		random_owned_province = {
			limit = {
				any_neighbor_province = {
					is_empty = yes
					tribal_land_of = ROOT
				}
			}
			random_empty_neighbor_province = {
				limit = {
					is_empty = yes
					tribal_land_of = ROOT
				}
				settle_province = ROOT
				if = {
					limit = { NOT = { culture = root religion = root } }
					change_culture = ROOT
					change_religion = ROOT
					add_devastation = 10
				}
			}
		}
		add_adm_power = -25
		add_dip_power = -25

	}
