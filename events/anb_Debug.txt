##Colonist stuck on settlement growth fix
#This is Heho code, we tried to change this. It didn't work. Lesson learned, don't touch Heho code.
country_event = {
	id = anb_debug.2
	title = anb_debug.2.t
	desc = anb_debug.2.d
	picture = HUIZTILOPOCHTLI_eventPicture
	
	hidden = yes
	
	mean_time_to_happen = {
		months = 6
	}
	
	trigger = {
		#has_country_flag = avoids_colonist_settlement_growth
		num_of_colonists = 1
		OR = {
			capital_scope = { has_colonist = yes } #This lower intensity and AI like putting their settler in the capital
			AND = {
				NOT = { num_of_colonies = 1 }
				ai = yes
				any_owned_province = {
					has_colonist = yes
				}
			}
		}
		has_dlc = "Dharma"
	}
	
	option = {
		name = debug.6.a
		
		every_owned_province = {
			limit = { has_colonist = yes is_city = yes }
			cancel_construction = yes
		}
		if = {	#Create a colony and don't take their colonist away if they need it
			limit = { 
				any_owned_province = { is_city = yes has_empty_adjacent_province = yes } #Don't take away their colonists if they need to make a colony
			}
			random_owned_province = {
				limit = { is_city = yes has_empty_adjacent_province = yes }
				random_empty_neighbor_province = { #colonize things downstream you dumbshits
					limit = { development = 3 }	#Don't colonize 0 0 0
					create_colony = 300	#Boosted or this colony will never finish cause the AI doesn't send their colonists
				}
			}
		}
		else_if = { #Monstrous frontiers stop AI doing good colonizing
			limit = {
				any_owned_province = {
					has_construction = siberian
				}
			}
			add_country_modifier = {
				name = dont_colonize_but_migrate
				duration = 1825
			}
		}
		else_if = { #Check if they still need their colonists
			limit = {
				NOT = { num_of_colonies = 1 }
			}
			if = { #turn off the colonist if all of escann is colonized
				limit = {
					capital_scope = {
						superregion = escann_superregion
					}
					escann_superregion = {
						is_city = yes
						type = all
					}
				}
				add_country_modifier = {	#Take away their colonist for the rest of the game
					name = dont_colonize_but_migrate
					duration = -1
				}
			}
			else_if = { #Turn off the colonist if all the serpentspine is colonized
				limit = {
					capital_scope = {
						continent = serpentspine
					}
					serpentspine = {
						is_city = yes
						type = all
					}
				}
				add_country_modifier = {	#Take away their colonist for the rest of the game
					name = dont_colonize_but_migrate
					duration = -1
				}
			}
			else_if = {
				any_owned_province = { has_port = yes }
				add_country_modifier = {	#Take away their colonist for a few years, they might need it again if they regain a coast
					name = dont_colonize_temp
					duration = 3650
				}
			}
			else = {
				add_country_modifier = {	#Take away their colonist for the rest of the game
					name = dont_colonize_but_migrate
					duration = -1
				}
			}
		}
	}
}

##Cleaning province
province_event = {
	id = anb_debug.3
	title = anb_debug.3.t
	desc = anb_debug.3.d
	picture = HUIZTILOPOCHTLI_eventPicture
	
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		has_province_flag = cant_colonize
	}
	
	option = {
		name = debug.6.a
		
		if = {
			limit = { 
				owner = { ai = no }
			}
			clr_province_flag = cant_colonize
		}
		else = {
			add_colonysize = -1000
			clr_province_flag = cant_colonize
		}
	}
}
