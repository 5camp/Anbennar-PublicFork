country_event = { 
	id = anb_debug.2 
	title = anb_debug.2.t 
	desc = anb_debug.2.d 
	picture = HUIZTILOPOCHTLI_eventPicture 
	 
	hidden = yes 
	 
	mean_time_to_happen = { 
		months = 6 
	} 
	 
	trigger = { 
		has_country_flag = free_the_colonists 
		any_owned_province = { 
			has_colonist = yes 
			is_city = yes 
		} 
		ai = yes 
	} 
	
	option = {
		name = debug.6.a
		if = {
			limit = {
				has_dlc = "Golden Century"
			}
			every_owned_province = {
				limit = { has_colonist = yes is_city = yes }
				cancel_construction = yes
			}
			random_empty_neighbor_province = {
				create_colony = 1
			}
			if = {
				limit = {
					capital_scope = {
						superregion = escann_superregion
					}
				}
				if = {
					limit = {
						NOT = { num_of_colonies = 1 }
						escann_superregion = {
							is_city = yes
							type = all
						}
					}
					add_country_modifier = {
						name = dont_colonize_but_migrate
						duration = -1
					}
				}
			}
			else_if = {
				limit = {
					capital_scope = {
						continent = serpentspine
					}
					serpentspine = {
						is_city = yes
						type = all
					}
				}
				add_country_modifier = {
					name = dont_colonize_but_migrate
					duration = -1
				}
			}
			if = {
				limit = {
					NOT = { num_of_colonies = 1 }
				}
				add_country_modifier = {
					name = dont_colonize_temp
					duration = 365
				}
			}
		}
	}
}

##Cleaning province
province_event = {
	id = anb_debug.3
	title = anb_debug.3.t
	desc = anb_debug.3.d
	picture = HUIZTILOPOCHTLI_eventPicture
	
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		has_province_flag = cant_colonize
	}
	
	option = {
		name = debug.6.a
		
		if = {
			limit = { 
				owner = { ai = no }
			}
			clr_province_flag = cant_colonize
		}
		else = {
			add_colonysize = -1000
			clr_province_flag = cant_colonize
		}
	}
}
