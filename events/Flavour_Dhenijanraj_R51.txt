country_event = {
	id = Flavour_R51_events.3
	title = "Flavour_R51_events.3.t"
	desc = "Flavour_R51_events.3.d"
	picture = RELIGIOUS_TURMOIL_eventPicture

	fire_only_once = yes

	trigger = {
		always = yes
	}

	is_triggered_only = yes

	option = {
		name = "Flavour_R51_events.3.a"
		add_treasury = -1000
		add_manpower = -10
		add_adm_power = -200
		if = {
			limit = {
				4265 = { owner = NAT }
			}
			4265 = {
			    cede_province = I18
			}
			4265 = {
			    add_core = I18
			}
			4265 = {
			    change_culture = peridot_dwarf
			}
			4265 = {
			    change_religion = high_philosophy
			}
			hidden_effect = {
				I18 = {
					change_religion = high_philosophy
					set_religious_school = {
    				group = raheni
    				school = orange_sash_school
					}
					define_ruler = {
	    			name = "Sindri 'the Returned'"
					    dynasty = "of the Peridot Flower"
					    adm = 5
					    dip = 4
					    mil = 4
					    age = 73
					    claim = 95
					}
				}
			}
	    create_subject = {
				subject_type = autonomous_vassal
				subject = I18
			}
		}
		else = {
			province_id = 4625 = {
				limit = {
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT}
				}
				add_permanent_claim = ROOT
				set_country_flag = dhenijanraj_grozumdihr_promise
			}
		}

		}

		option = {
			name = "Flavour_R51_events.3.b"
			large_decrease_of_dwarven_tolerance_effect = yes
	}
}
