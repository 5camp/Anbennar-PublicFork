#Glowing City
province_event = {
	id = fragments.1
	title = fragments.1.t
	desc = fragments.1.d
	picture = ARTWORK_eventPicture
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	trigger = {
		has_dlc = "El Dorado"
		OR = {
			continent = north_america
			continent = south_america
		}
		FROM = {
			NOT = {
				has_country_flag = pf_chain_flag
				has_country_flag = pf_end_glowing_city_flag
			}
		}
		NOT = { has_global_flag = pf_end_glowing_city_flag_global }
	}
	
	#Launch the Search
	option = {
        name = fragments.1.a
        ai_chance = { factor = 100 }
		FROM = {
			hidden_effect = { set_country_flag = pf_chain_flag }
			hidden_effect = { set_variable = { searchPoints = 0 } }
			add_country_modifier = {
				name = pf_glowing_city
				duration = -1
			}
		}
	}
	#Rumours
	option = {
        name = fragments.1.b
        ai_chance = { factor = 0 }
		FROM = { hidden_effect = { set_country_flag = pf_end_glowing_city_flag } }
	}
}
