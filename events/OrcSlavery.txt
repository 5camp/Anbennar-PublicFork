province_event = { # Orcish slaves in Aelantir
	id = orc_slavery.1
	title = orc_slavery.1.t
	desc = orc_slavery.1.d
	picture = WESTERNISATION_eventPicture
	
	trigger = {
		OR = {
			continent = south_america
			continent = north_america
		}
		owner = {
			technology_group = tech_cannorian
		}
		NOT = { has_province_flag = orc_slave_workforce }
		OR = {
			province_with_farm_goods = yes
			province_with_mineable_goods = yes
		}
		NOT = {
			OR = {
				trade_goods = cloth
				trade_goods = paper
				trade_goods = glass
				trade_goods = chinaware
				trade_goods = slaves
			}
		}
	}

	mean_time_to_happen = {
		years = 5
		
		modifier = {
			factor = 0.75
			OR = {
				trade_goods = iron
				trade_goods = copper
				trade_goods = gold
				trade_goods = gems
			}
		}
		modifier = {
			factor = 2
			unrest = 2
		}
		modifier = {
			factor = 2
			unrest = 4
		}
		modifier = {
			factor = 2
			unrest = 6
		}
		modifier = {
			factor = 2
			unrest = 8
		}
		modifier = {
			factor = 5
			unrest = 10
		}
	}

	option = { # 
		name = orc_slavery.1.a
		ai_chance = {
			factor = 1
		}
		add_orcish_minority_size_effect = yes
		set_province_flag = orc_slave_workforce
		add_permanent_province_modifier = {
			name = orcish_slaves
			duration = 18250
		}
		random_list = {
			33 = {  }
			33 = { add_base_tax = 1 }
			33 = { add_base_production = 1 }
		}
	}
}
