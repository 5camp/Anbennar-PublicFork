		every_owned_province = {
			limit = {
				OR = {
					culture = khom
					is_core = Y66
				}
			}
			cede_province = Y67
		}
		every_subject_country = {
			limit = {
				NOT = { tag = Y70 }
			}
			every_owned_province = {
				limit = {
					OR = {
						is_core = Y66
						is_core = Y67
					}
					OR = {
						culture = khom
						culture = phonan
					}
				}
				cede_province = Y67
			}
		}

		#releases phonan
		if = {
			limit = { has_country_flag = Y63_elephant_grove_flag }
			every_owned_province = {
				limit = {
					is_core = Y70
					NOT = { is_capital = yes }
				}
				cede_province = Y70
			}
			every_subject_country = {
				limit = {
					any_owned_province = { culture = phonan }
				}

					limit = { culture = phonan }
					cede_province = Y70
				}
			}
			create_subject = {
			    subject_type = patronate
			    subject = Y70
			}

	}
