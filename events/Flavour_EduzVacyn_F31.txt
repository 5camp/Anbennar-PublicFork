#Plea or Demand
country_event = {
	id = flavor_vacyn.2
	title = flavor_vacyn.2.t
	desc = flavor_vacyn.2.d
	picture = KING_SICK_IN_BED_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		tag = F31
	}
	
	option = {
		name = flavor_vacyn.2
		ai_chance = { factor = 1 }

		if = {
			limit = {
				estate_loyalty = {
					estate = estate_nobility
					loyalty = 40
				}
			}
			east_naza_area = {
				spawn_rebels = {
					type = noble_rebels
					size = 2
				}
				add_province_modifier = {
					name = vacyntassi_plea_or_demand_strong_unrest_modifier
					duration = 5475
				}
			}
			south_naza_area = {
				spawn_rebels = {
					type = noble_rebels
					size = 2
				}
				add_province_modifier = {
					name = vacyntassi_plea_or_demand_strong_unrest_modifier
					duration = 5475
				}
			}
		}
		else_if = {
			limit = {
				estate_loyalty = {
					estate = estate_nobility
					loyalty = 50
				}
			}
			east_naza_area = {
				spawn_rebels = {
					type = noble_rebels
					size = 1
				}
				add_province_modifier = {
					name = vacyntassi_plea_or_demand_weak_unrest_modifier
					duration = 5475
				}
			}
			south_naza_area = {
				spawn_rebels = {
					type = noble_rebels
					size = 1
				}
				add_province_modifier = {
					name = vacyntassi_plea_or_demand_weak_unrest_modifier
					duration = 5475
				}
			}
		}
		else = {
			east_naza_area = {
				add_province_modifier = {
					name = vacyntassi_plea_or_demand_production_modifier
					duration = 5475
				}
			}
			south_naza_area = {
				add_province_modifier = {
					name = vacyntassi_plea_or_demand_production_modifier
					duration = 5475
				}
			}
		}
	}
}
