#The Ashen Ralkadezim
country_event = {
	id = flavor_vacyn.2
	title = flavor_vacyn.2.t
	desc = flavor_vacyn.2.d
	picture = KING_SICK_IN_BED_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		tag = F31
	}
	
	option = {
		name = flavor_vacyn.2
		ai_chance = { factor = 1 }

		if = {
			limit = {
				estate_loyalty = {
					estate = estate_nobility
					loyalty = 60
				}
			}
			east_naza_area = {
				add_province_modifier = {
					name = vacyntassi_ashen_ralkadezim_good_modifier
					duration = 5475
				}
			}
			south_naza_area = {
				add_province_modifier = {
					name = vacyntassi_ashen_ralkadezim_good_modifier
					duration = 5475
				}
			}
			south_naza_area = {
				add_province_modifier = {
					name = vacyntassi_ashen_ralkadezim_good_modifier
					duration = 5475
				}
			}
		}
		else = {
			change_estate_land_share = {
				estate = estate_nobles
				share = -15
			}
			east_naza_area = {
				spawn_rebels = {
					type = noble_rebels
					size = 2
				}
				add_province_modifier = {
					name = vacyntassi_ashen_ralkadezim_bad_modifier
					duration = 5475
				}
			}
			south_naza_area = {
				spawn_rebels = {
					type = noble_rebels
					size = 2
				}
				add_province_modifier = {
					name = vacyntassi_ashen_ralkadezim_bad_modifier
					duration = 5475
				}
			}
		}
	}
}
