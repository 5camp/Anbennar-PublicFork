country_event = {
	id = verne.12
	title = verne.12.t
	desc = verne.12.d
	picture = RELIGION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = A33
			
		is_religion_enabled = corinite
		
		NOT = { religion = corinite }
	}
	
	mean_time_to_happen = {
		months = 100
		
		modifier = {
			factor = 0.5
			any_owned_province = { 
				religion = corinite
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			capital_scope = {
				every_province_in_state = {
					change_religion = corinite
				}
			}
			every_owned_province = {
				limit = { has_estate = estate_adventurers }
				change_religion = corinite
			}
		}
	}
	
	#Convert to Corinite
	option = {
		name = verne.12.a
		ai_chance = {
			factor = 80
			modifier = {
				factor = 3
				any_owned_province = {
				religion = corinite
				}
			}
			modifier = {
				factor = 0.5
				is_emperor = yes
			}
			modifier = {
				factor = 3
				any_owned_province = {
				is_reformation_center = yes
				}
			}
			modifier = {
				factor = 1.5
				has_personal_deity = corin
			}
			modifier = {
				factor = 1.5
				mil = 4
			}
			modifier = {
				factor = 0
				is_emperor = yes
			}
			modifier = {
				factor = 1.5
				personality = ai_militarist
			}
			modifier = {
				factor = 1.25
				ruler_has_personality = inspiring_leader_personality
			}
			modifier = {
				factor = 1.25
				ruler_has_personality = bold_fighter_personality
			}
		}
		capital_scope = {
			change_religion = corinite
			add_reform_center = corinite
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
		}
		change_religion = corinite
	}
	
	#Stay Regent Court
	option = {
		name = verne.12.b
		ai_chance = {
			factor = 20
			modifier = {
				factor = 1.25
				stability = 3
			}
			modifier = {
				factor = 1.25
				has_personal_deity = adean
			}
			modifier = {
				factor = 1.25
				ruler_has_personality = craven_personality
			}
			modifier = {
				factor = 1.25
				ruler_has_personality = careful_personality
			}
			modifier = {
				factor = 2
				is_defender_of_faith = yes
			}
		}
		add_stability = -1
	}
