#The Arrival of the Young Owl
country_event = {
	id = flavor_rogieria.2
	title = flavor_rogieria.2.t
	desc = flavor_rogieria.2.d
	picture = CITY_VIEW_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		tag = B37
	}
	
	immediate = {
		hidden_effect = {
			set_variable = { which = RogieriaWitchKingPath value = 0 }
			set_variable = { which = RogieriaLegacyPath value = 0 }
			set_variable = { which = RogieriaGoodGuyPath value = 0 }
			set_variable = { which = RogieriaEmulantAdenn value = 0 }
		}
	}
	
	#Rogier will lead us to victory
	option = {
		name = flavor_rogieria.2.a
		ai_chance = { factor = 1 }
		define_ruler = {
			name = "Rogier 'Young Owl'"
			dynasty = "Silmuna"
			claim = 100
			culture = half_orc
			age = 26
			male = yes
			adm = 5
			dip = 4
			mil = 4
		}
		add_ruler_personality = silver_tongue_personality
		swap_non_generic_missions = yes
		remove_heir = {}
	}
}
