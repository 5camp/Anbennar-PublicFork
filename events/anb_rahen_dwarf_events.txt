		current_age = { age_of_reformation }

			}
		}
		num_owned_provinces = {
			region = tree_of_stone_region
			value = 7
		}
		OR = {

			has_dwarven_minority_trigger = yes
		}

			}
		}
		else = {
			changevariable = {
				which = rahen_dwarf_exodus
				value = 1
			}
		}

        name = anb_rahen_dwarfevents.1.a
