country_event = {
	id = flavour_plumstead.26
	title = flavour_plumstead.26.t
	desc = {
		trigger = { has_country_flag = H61_vassal_peaceful }
		desc = flavour_plumstead.26.da
	}
	desc = {
		trigger = { has_country_flag = H61_vassal_spies }
		desc = flavour_plumstead.26.db
	}
	picture = COURT_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	option = {
		name = "flavour_plumstead.26.a"
		ai_chance = { factor = 100 } #TODO decide AI chance
		FROM = {
			hidden_effect = {
				set_country_flag = H61_accept_vassal
			}
			vassalize = ROOT
			country_event = { flavour_plumstead.27 }
		}
		trollsmaw_swamp_area = {
			add_permanent_claim = ROOT
		}
		coast_of_isobel_area = {
			add_permanent_claim = ROOT
		}
		boek_river_area = {
			add_permanent_claim = ROOT
		}
		thilvis_area = {
			add_permanent_claim = ROOT
		}
		ikla_area = {
			add_permanent_claim = ROOT
		}
		zanlib_area = {
			add_permanent_claim = ROOT
		}
	}
	
	option = {
		name = "flavour_plumstead.26.b"
		ai_chance = { factor = 100 }
		FROM = {
			hidden_effect = {
				set_country_flag = H61_refuse_vassal
			}
			country_event = { flavour_plumstead.27 }
			tooltip = {
				add_casus_belli = {
					target = ROOT
					type = cb_vassalize_mission
					months = 180
				}
			}
		}
		hidden_effect = {
			set_country_flag = H61_refuse_vassal
		}
	}
	
	after = {
		clr_country_flag = H61_vassal_peaceful
		clr_country_flag = H61_vassal_spies
	}
}
