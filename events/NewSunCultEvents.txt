#Lucian Carter arrives in the Drolas (Betelis province)
province_event = {
	id = new_sun_cult.9
	title = new_sun_cult.9.t
	desc = new_sun_cult.9.d
	picture = DIPLOMACY_eventPicture
	goto = ROOT
	
	is_triggered_only = yes
	fire_only_once = yes
	
	immediate = {
		random_province = {
			limit = {
				OR = {
					area = inner_drolas_area
					area = outer_drolas_area
				}
				NOT = { has_province_modifier = nsc_teaching_the_one_god }
			}
			save_event_target_as = drolas_province_1
		}
		random_province = {
			limit = {
				OR = {
					area = inner_drolas_area
					area = outer_drolas_area
				}
				NOT = { has_province_modifier = nsc_teaching_the_one_god }
			}
			save_event_target_as = drolas_province_2
		}
	}
	
	#Leave him
	option = {
		name = new_sun_cult.9.a
		ai_chance = { factor = 100 }
		change_religion = ravelian
		add_permanent_province_modifier = { name = "religious_zeal_at_conv" duration = 9125 }
		add_permanent_province_modifier = { name = nsc_teaching_the_one_god duration = 9125 }
		event_target:drolas_province_1 = {
			change_religion = ravelian
			add_permanent_province_modifier = { name = nsc_teaching_the_one_god duration = 9125 }
			add_permanent_province_modifier = { name = "religious_zeal_at_conv" duration = 9125 }
		}
		event_target:drolas_province_2 = {
			change_religion = ravelian
			add_permanent_province_modifier = { name = nsc_teaching_the_one_god duration = 9125 }
			add_permanent_province_modifier = { name = "religious_zeal_at_conv" duration = 9125 }
		}
		set_country_flag = nsc_lucian_carter_in_our_lands
	}
	
	#Expel the dude
	option = {
		name = new_sun_cult.9.b
		ai_chance = { factor = 0 }
		add_adm_power = -50
		change_religion = ravelian
		add_permanent_province_modifier = { name = nsc_drolas_missionaries_expelled duration = 3650 }
		event_target:drolas_province_1 = {
			change_religion = ravelian
			add_permanent_province_modifier = { name = nsc_drolas_missionaries_expelled duration = 3650 }
		}
		event_target:drolas_province_2 = {
			change_religion = ravelian
			add_permanent_province_modifier = { name = nsc_drolas_missionaries_expelled duration = 3650 }
		}
	}
	
	after = {
		set_country_flag = nsc_lucian_carter_happened
		every_country = {
			limit = {
				religion = bulwari_sun_cult
				NOT = { capital_scope = { continent = south_america } }
				NOT = { capital_scope = { continent = north_america } }
			}
			set_country_flag = nsc_lucian_carter_happened
		}
	}
}

		}
		if = { 
			limit = {
				REB = {
					OR = {
						check_variable = { which = nscChoiceAVar value = 66 }

			}

						check_variable = { which = nscChoiceCVar value = 33 }
					}
				}
			}
			custom_tooltip = nsc_small_increase_unity_same_tt
		}
		else {
			custom_tooltip = nsc_stagnating_unity_tt
		}
		clr_country_flag = nsc_choice_a
		clr_country_flag = nsc_choice_b
		clr_country_flag = nsc_choice_c
	}
