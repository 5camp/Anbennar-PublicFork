province_event = {
	id = flavour_arawkelin.10
	title = flavour_arawkelin.10.t
	desc = flavour_arawkelin.10.d
	
	picture = COURT_eventPicture
	
	hidden = yes
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				OR = {
					NOT = {
						has_province_modifier = minor_deforestation
						has_province_modifier = major_deforestation
						has_province_modifier = full_deforestation
						has_province_modifier = soil_erosion
					}
				}
			}
			add_permanent_province_modifier = {
				name = minor_deforestation
				duration = -1
			}
			province_event = {
				id = flavour_arawkelin.10
				days = 3652
			}
		}
		else_if = {
			limit = {
				has_province_modifier = minor_deforestation
			}
			remove_province_modifier = minor_deforestation
			add_permanent_province_modifier = {
				name = major_deforestation
				duration = -1
			}
			province_event = {
				id = flavour_arawkelin.10
				days = 3652
			}
		}
		else_if = {
			limit = {
				has_province_modifier = major_deforestation
			}
			remove_province_modifier = major_deforestation
			add_permanent_province_modifier = {
				name = full_deforestation
				duration = -1
			}
			province_event = {
				id = flavour_arawkelin.10
				days = 18262
			}
		}
		else_if = {
			limit = {
				has_province_modifier = full_deforestation
				NOT = {
					has_country_flag = unlocked_artifice_invention_growth_beans
				}
			}
			remove_province_modifier = full_deforestation
			add_permanent_province_modifier = {
				name = soil_erosion
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = full_deforestation
				has_country_flag = unlocked_artifice_invention_growth_beans
			}
			remove_province_modifier = major_deforestation
			add_permanent_province_modifier = {
				name = artificial_farmland
				duration = -1
			}
		}
	}
	
	option = {
		flavour_arawkelin.10.a
	}
}
