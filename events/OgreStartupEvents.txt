#Reign of Ziltagh
country_event = {
	id = ogre.1
	title = ogre.1.t
	picture = DIPLOMACY_SIGNING_eventPicture
	desc = ogre.1.d
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		is_year = 1444
		NOT = { has_country_flag = flag_of_ziltagh }
		tag = I90
	}
	
	immediate = {
		set_country_flag = flag_of_ziltagh
	}
	
	#Long may he Reign
	option = {
		name = ogre.1.a
		ai_chance = { factor = 100 }
		add_ruler_modifier = { 
			name = reign_of_ziltagh
			duration = -1
		}
	}
}
