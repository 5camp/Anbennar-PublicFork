# Deepening the capital
country_event = {
    id = owb.2
    title = owb.2.t
    desc = owb.2.d
    picture = ARTWORK_eventPicture
    
    is_triggered_only = yes
	
    trigger = {
		always = yes
	}
	
	immediate = {
		hidden_effect = { capital_scope = { setup_vault_digging_vault = yes } }
	}
	
	after = {
		hidden_effect = { clr_country_flag = debug_vault_flag }
	}
	
    #Let's dig
    option = {
        name = owb.2.a
		if = {
			limit = {
				capital_scope = {
					NOT = { has_construction = building }
					NOT = { has_construction = great_project }
					NOT = { has_construction = canal }
					NOT = { has_construction = culture }
					NOT = { has_construction = missionary }
					NOT = { has_construction = army }
					NOT = { has_construction = navy } 
				}
			}
			capital_scope = { start_vault_digging = yes }
			hidden_effect = { set_country_flag = vault_digging }
		}
		else = {
			custom_tooltip = diggy_no_building_event
		}
    }
    
    #Go back
    option = {
		ai_chance = { factor = 10 }
        name = owb.2.b
    }
}

#Study Center
country_event = {
    id = owb.10
    title = owb.10.t
    desc = owb.10.d
    picture = ARTWORK_eventPicture
    
    is_triggered_only = yes
	
    trigger = {
		tag = OWB
	}
	
	immediate = {
		hidden_effect = {
			while = {
				limit = { NOT = { check_variable = { foo = 35 } } }
				change_variable = { foo = 1 }
				random_province = {
					limit = {
						OR = {
							continent = south_america
							continent = north_america
						}
						NOT = { has_province_modifier = owb_ancient_precursor_artefact }
						is_wasteland = no
					}
					add_province_triggered_modifier = owb_ancient_precursor_artefact
				}
			}
			#Glowing city
			random_province = {
				limit = {
					is_wasteland = no
					OR = {
						continent = south_america
						continent = north_america
					}
				}
				set_province_flag = owb_glowing
				add_province_triggered_modifier = owb_precursor_fragment
			}
			#Dragon House
			random_province = {
				limit = {
					is_wasteland = no
					region = amadia_region
				}
				set_province_flag = owb_dragon_house
				add_province_triggered_modifier = owb_precursor_fragment
			}
			#Elfenbride's Mansion
			random_province = {
				limit = {
					is_wasteland = no
					OR = { region = rzenta_region region = forest_of_the_cursed_ones_region region = dalaire_region }
				}
				set_province_flag = owb_elfenbride
				add_province_triggered_modifier = owb_precursor_fragment
			}
			#God fragment
			random_province = {
				limit = {
					is_wasteland = no
					colonial_region = colonial_noruin
				}
				set_province_flag = owb_god_fragment
				add_province_triggered_modifier = owb_precursor_fragment
			}
			#Vault
			random_province = {
				limit = {
					is_wasteland = no
					OR = { colonial_region = colonial_upper_ynn colonial_region = colonial_lower_ynn colonial_region = colonial_expanse }
				}
				set_province_flag = owb_vault
				add_province_triggered_modifier = owb_precursor_fragment
			}
			#Portal Nexus
			random_province = {
				limit = {
					is_wasteland = no
					OR = { continent = south_america continent = north_america }
				}
				set_province_flag = owb_portal
				add_province_triggered_modifier = owb_precursor_fragment
			}
			#Floating City
			random_province = {
				limit = {
					is_wasteland = no
					OR = { region = west_effelai_region region = east_effelai_region region = south_effelai_region region = leechdens_region }
				}
				set_province_flag = owb_floating
				add_province_triggered_modifier = owb_precursor_fragment
			}
			set_variable = { foo = 0 }
		}
	}
	
	after = {
	}
	
    #Dope
    option = {
        name = owb.10.a
		custom_tooltip = owb_we_can_find_apa_tooltip
    }
}
