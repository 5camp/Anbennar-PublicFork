#Emperor gets demand
country_event = {
	id = flavor_roilsard.6
	title = flavor_roilsard.6.t
	desc = flavor_roilsard.6.d
	picture = TRADEGOODS_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		always = yes
	}
	
	#Yes
	option = {
		name = flavor_roilsard.6.a
		FROM = { country_event = { id = flavor_roilsard.7 days = 10 random = 8 } }
		FROM = {
			elector = yes
			set_government_rank = 2
		}
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				emperor = { is_rival = FROM }
			}
		}
	}
	
	#Fuk dem
	option = {
		name = flavor_roilsard.6.b
		FROM = { country_event = { id = flavor_roilsard.8 days = 10 random = 8 } }
		add_stability = -2
		add_imperial_influence = -50
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				NOT = { emperor = { is_rival = FROM } }
			}
		}
	}
	
}
