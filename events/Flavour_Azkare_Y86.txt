#Kai
country_event = {
	id = azkare.101
	title = azkare.101.t
	desc = azkare.101.d
	picture = EMPTY_eventPicture
	
	is_triggered_only = yes
	
	#Next Page
	option = {
		name = azkare.101.a
		hidden_effect = { country_event = { id = azkare.102 } }
	}
	
	#Previous Page
	option = {
		name = azkare.101.b
		hidden_effect = { country_event = { id = azkare.110 } }
	}
	
	#Reorganise
	option = {
		name = azkare.101.c
		if = {
			limit = {
				5429 = {
					OR = {
						NOT = { has_province_flag = azkare_reorganisation }
						had_province_flag = { flag = azkare_reorganisation days = 18250 }
					}
				}
			}
			add_country_modifier = { name = azkare_reorganise_kai duration = 3650 }
			5429 = { set_province_flag = azkare_reorganisation }
			custom_tooltip = azkare_reorganise_kai_tt
			hidden_effect = {
				every_owned_province = {
					limit = { culture_group = kai }
					set_seat_in_parliament = no
				}
				country_event = { id = azkare.100 }
			}
		}
		else = {
			custom_tooltip = azkare_reorganise_unavailable_tt
			hidden_effect = { country_event = { id = azkare.100 } }
		}
	}
	
	#Back to Menu
	option = {
		name = azkare.101.e
		hidden_effect = { country_event = { id = azkare.100 } }
	}
}
