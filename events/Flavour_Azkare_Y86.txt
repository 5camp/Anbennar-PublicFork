	desc = { trigger = { is_variable_equal = { which = azkareParliamentMenu value = 10 } } desc = azkare.100.dj }
	picture = EMPTY_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			azkare_parliament_calculations = yes
		}
	}
	
	#Next Page
	option = {
		name = azkare.100.a
		hidden_effect = {
			change_variable = { which = azkareParliamentMenu value = 1 }
			if = { limit = { check_variable = { which = azkareParliamentMenu value = 11 } }
				set_variable = { which = azkareParliamentMenu value = 1 }
			}
			country_event = { id = azkare.101 }
		}
	}
	
	#Previous Page
	option = {
		name = azkare.100.b
		hidden_effect = {
			change_variable = { which = azkareParliamentMenu value = -1 }
			if = { limit = { NOT = { check_variable = { which = azkareParliamentMenu value = 1 } } }
				set_variable = { which = azkareParliamentMenu value = 10 }
			}
			country_event = { id = azkare.101 }
		}
	}
	
	#Close
	option = {
		name = azkare.100.c
		clr_country_flag = azkare_parliament_menu
	}
}
