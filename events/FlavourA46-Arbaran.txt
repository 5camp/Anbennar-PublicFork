#Arbaran's Escandar event chain
#Has three different ways to get it back, buying it, threatening the owner, just declaring war

namespace = arbaran

country_event = {
	id = arbaran.1 #Initial Choice
	title = arbaran.1.t
	desc = arbaran.1.d
	picture = FORT_eventPicture
	
	is_triggered_only = yes
	
	option = { #Let's try to buy it
		name = arbaran.1.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				treasury = 5000
			}
			modifier = {
				factor = 1.5
				treasury = 10000
			}
			modifier = {
				factor = 0.1
				316 = {
					owner = {
						is_rival = ROOT
					}
				}
			}
		}
		country_event = { id = arbaran.2 days = 1 }
	}
	
	option = { #Threaten them
		name = arbaran.1.b
		ai_chance = {
			factor = 40
			modifier = {
				factor = 1.5
				NOT = { treasury = 5000 }
			}
			modifier = {
				factor = 3
				316 = {
					owner = {
						is_rival = ROOT
					}
				}
			}
		}
		316 = {
			owner = {
				country_event = { id = arbaran.5 days = 14 random = 7 }
			}
		}
	}
	
	option = { #War
		name = arbaran.1.c
		ai_chance = {
			factor = 10
		}
		hidden_effect = {
			316 = {
				owner = { save_event_target_as = escandar_owner }
			}
		}
		declare_war_with_cb = {
			who = 316
			casus_belli = cb_conquest
			war_goal_province = 316
		}
	}
}

country_event = {
	id = arbaran.3 #They receive our offer of trade
	title = arbaran.3.t
	desc = {
		trigger = { has_global_flag = arbaran_low_offer }
		desc = arbaran.3.dlow
	}
	
	desc = {
		trigger = { has_global_flag = arbaran_medium_offer }
		desc = arbaran.3.dmedium
	}
	
	desc = {
		trigger = { has_global_flag = arbaran_high_offer }
		desc = arbaran.3.dhigh
	}
	
	picture = FORT_eventPicture
	
	is_triggered_only = yes
	
	option = { #Aight
		name = arbaran.3.a
		ai_chance = {
			factor = 100 #CHANGE
		}
		A46 = { country_event = { id = arbaran.7 days = 14 random = 7 } } #THEY ACCEPT EVENT ID HERE
	}
	
	option = { #Nah
		name = arbaran.3.b
		ai_chance = {
			factor = 0		#CHANGE
			}
		A46 = { country_event = { id = arbaran.4 days = 14 random = 7 } } #REFUSAL TO SELL ID HERE
	}
}

country_event = {
	id = arbaran.4 #They refused our offer of money. What now?
	title = arbaran.4.t
	desc = arbaran.4.d
	picture = FORT_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		clr_global_flag = arbaran_low_offer
		clr_global_flag = arbaran_medium_offer
		clr_global_flag = arbaran_high_offer
	}
	
	option = { #Time to threaten them
		name = arbaran.4.a
		316 = {
			owner = {
				country_event = { id = arbaran.5 days = 14 random = 7 }
			}
		}
	}
	
	option = { #DECLARE WAR
		name = arbaran.4.b
		hidden_effect = {
			316 = {
				owner = { save_event_target_as = escandar_owner }
			}
		}
		declare_war_with_cb = {
			who = event_target:escandar_owner
			casus_belli = cb_conquest
			war_goal_province = 316
		}
	}
	
	option = { #Drop the matter for now
		name = arbaran.4.c
		add_prestige = -25
	}
}

country_event = {
	id = arbaran.7 #They accepted the trade!
	title = arbaran.7.t
	desc = arbaran.7.d
	picture = FORT_eventPicture
	
	is_triggered_only = yes
	
	option = { #Epic!
		name = arbaran.7.a
		if = {
			limit = { has_global_flag = arbaran_low_offer }
			add_treasury = -1000
		}
		if = {
			limit = { has_global_flag = arbaran_medium_offer }
			add_treasury = -2500
		}
		if = {
			limit = { has_global_flag = arbaran_high_offer }
			add_treasury = -5000
		}
		316 = {
			owner = {
				if = {
					limit = { has_global_flag = arbaran_low_offer }
					add_treasury = 1000
				}
				if = {
					limit = { has_global_flag = arbaran_medium_offer }
					add_treasury = 2500
				}
				if = {
					limit = { has_global_flag = arbaran_high_offer }
					add_treasury = 5000
				}
			}
		}
		316 = {
			cede_province = A46
			add_core = A46
		}
		clr_global_flag = arbaran_low_offer
		clr_global_flag = arbaran_medium_offer
		clr_global_flag = arbaran_high_offer
	}
}
