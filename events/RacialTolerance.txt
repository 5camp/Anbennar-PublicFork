		}
		#Greater chance to happen the larger and more oppressed the people

		}
		modifier = {

			has_province_modifier = elven_minority_oppressed_small
		}
		modifier = {
			factor = 2
			has_province_modifier = elven_minority_oppressed_large
		}
	}
	
	option = {		# Of course!
		name = "racial_tolerace.1.a"
		large_increase_of_elven_tolerance_effect = yes
		add_local_autonomy = 20
	}
	option = {		# Make some concessions
		name = "racial_tolerace.1.b"
		small_increase_of_elven_tolerance_effect = yes
		add_local_autonomy = 20
	}

		name = "racial_tolerace.1.c"
		large_decrease_of_elven_tolerance_effect = yes
		add_local_autonomy = 20
	}
