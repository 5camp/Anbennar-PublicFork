#Removing Damestear
province_event = {
	id = damestear.2
	title = damestear.2.t
	desc = damestear.2.d
	picture = TRADE_GOODS_METALS_AND_MINING_EQUIPENT_eventPicture
	
	mean_time_to_happen  = {
		years = 40
		
		modifier = {
			factor = 0.7
			REB = {
				check_variable = {
					which = num_damestear
					value = 15
				}
			}
		}
	}
	
	trigger = {
		trade_goods = damestear
	}
	
	option = {
        name = damestear.2.a
        ai_chance = { factor = 100 }
		ROOT = {
			change_trade_goods = unknown
			remove_province_modifier = damestear_rich_deposit
		}
		hidden_effect = {
			REB = {
				subtract_variable = {
					which = num_damestear
					value = 1
				}
			}
			set_rem_damestear_count = yes
		}
	}
}
