#The Response
country_event = {
	id = ogre.9
	title = ogre.9.t
	picture = BIG_BOOK_eventPicture
	desc = ogre.9.d
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		always = yes
	}
	
	#Let's see
	option = {
		name = ogre.9.a
		ai_chance = { factor = 100 }
		every_country = { limit = { has_country_flag = ogre_vassal }
			ROOT = {
				create_subject = {
					subject_type = tributary_state
					subject = PREV
				}
			}
		}
		every_country = { limit = { has_country_flag = ogre_refuse }
			ROOT = {
				add_casus_belli = {
					target = PREV
					type = cb_force_tributary
					months = 300
				}
			}
		}
	}
}

#Trade Goods Change
country_event = {
	id = ogre.10
	title = ogre.10.t
	picture = TRADE_GOODS_METALS_AND_MINING_EQUIPENT_eventPicture
	desc = ogre.10.d
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		always = yes
	}
	
	#Digging an Ogrehole
	option = {
		name = ogre.10.a
		ai_chance = { factor = 100 }
		random_list = {
			30 = {
				capital_scope = { change_trade_goods = copper }
			}
			25 = {
				capital_scope = { change_trade_goods = iron }
			}
			20 = {
				capital_scope = { change_trade_goods = gems }
			}
			20 = {
				capital_scope = { change_trade_goods = gold }
			}
			4 = {
				capital_scope = { change_trade_goods = damestear }
			}
			1 = {
				capital_scope = { 
					change_trade_goods = damestear
					set_province_flag = permanent_damestear
				}	
			}
		}
	}
}
