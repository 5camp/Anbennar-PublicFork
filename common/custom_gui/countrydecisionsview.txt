#Unique Tab

	name = artificery_unique_tab
	tooltip = artificery_unique_selection_tt
	potential = { artifice_invention_has_any_menu_open = yes }
	frame = {
		number = 1 
		trigger = {
			NOT = {
				OR = {
					has_country_flag = artifice_invention_menu_open_unique_large
					has_country_flag = artifice_invention_menu_open_unique_small
				}
			}
		}
	}
	frame = {
		number = 2
		trigger = {
			OR = {
				has_country_flag = artifice_invention_menu_open_unique_large
				has_country_flag = artifice_invention_menu_open_unique_small
			}
		}
	}
	trigger = { }
	effect = { 
		artifice_gui_tab_flag_clearing = yes
		if = {
			limit = {
				artifice_invention_is_unique_tab_large_qualifier = yes
			}
			set_country_flag = artifice_invention_menu_open_unique_large
		}
		else = {
			set_country_flag = artifice_invention_menu_open_unique_small
		}
	}
}
