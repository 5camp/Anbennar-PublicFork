custom_button = {
	name = artifice_open_inventions
	potential = {
		NOT = { has_country_flag = artifice_invention_menu_open } 
	}

	trigger = {
		OR = {
			has_country_flag = artifice_research_fast
			has_country_flag = artifice_research_normal

		}
	}
	
	effect = {
		set_country_flag = artifice_invention_menu_open
		set_country_flag = artificery_tab_generic_tier_1_open
	}

	tooltip = artifice_research_button_tt
}
