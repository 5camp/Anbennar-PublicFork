#Invention Buttons Tier 1
#Economic Tier 1
custom_button = {
	name = economic_invention_1_1
	tooltip = x
	potential = { 
		has_country_flag = artificery_tab_generic_tier_1_open
	}
	frame = {
		number = 1
		trigger = {
			NOT = {
				has_country_flag = unlocked_artifice_invention_artificer_exo_arms
			}
		}
	}
	frame = {
		number = 2
		trigger = {
			NOT = {
				has_estate_privilege = artifice_invention_artificer_exo_arms
			}
			has_country_flag = unlocked_artifice_invention_artificer_exo_arms
		}
	}
	frame = {
		number = 3
		trigger = {
			has_estate_privilege = artifice_invention_artificer_exo_arms
		}
	}
	trigger = {
		has_country_flag = unlocked_artifice_invention_artificer_exo_arms
	}
	effect = { 
		if = {
			limit = {
				artifice_can_select_invention_basic = yes
				NOT = {
					has_estate_privilege = artifice_invention_artificer_exo_arms
				}
			}
			set_estate_privilege = artifice_invention_artificer_exo_arms
		}
		else_if = {
			limit = {
				artifice_can_select_invention_basic = no
				NOT = {
					has_estate_privilege = artifice_invention_artificer_exo_arms
				}
			}
			custom_tooltip = artifice_cannot_afford_invention
			tooltip = {
				set_estate_privilege = artifice_invention_artificer_exo_arms
			}
		}
        else_if = {
			limit = {
				has_estate_privilege = artifice_invention_artificer_exo_arms
			}
			remove_estate_privilege = artifice_invention_artificer_exo_arms
		}
	}
}

#Society Tier 3
custom_button = {
	name = society_invention_3_1
	tooltip = x
	potential = { 
		has_country_flag = artificery_tab_generic_tier_3_open
	}
	frame = {
		number = 1
		trigger = {
			NOT = {
				has_country_flag = unlocked_artifice_invention_crierless_crier_device
			}
		}
	}
	frame = {
		number = 2
		trigger = {
			NOT = {
				has_estate_privilege = artifice_invention_crierless_crier_device
			}
			has_country_flag = unlocked_artifice_invention_crierless_crier_device
		}
	}
	frame = {
		number = 3
		trigger = {
			has_estate_privilege = artifice_invention_crierless_crier_device
		}
	}
	trigger = {
		has_country_flag = unlocked_artifice_invention_crierless_crier_device
	}
	effect = { 
		if = {
			limit = {
				artifice_can_select_invention_expert = yes
			}
			set_estate_privilege = artifice_invention_crierless_crier_device
		}
		else_if = {
			limit = {
				artifice_can_select_invention_expert = no
				NOT = {
					has_estate_privilege = artifice_invention_crierless_crier_device
				}
			}
			custom_tooltip = artifice_cannot_afford_invention
			tooltip = {
				set_estate_privilege = artifice_invention_crierless_crier_device
			}
		}
        else_if = {
			limit = {
				has_estate_privilege = artifice_invention_crierless_crier_device
			}
			remove_estate_privilege = artifice_invention_crierless_crier_device
		}
	}
}
