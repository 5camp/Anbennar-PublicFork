######################################################################
# Blood lotus rebel
######################################################################
blood_lotus_rebel = {

	color = { 50 85 210 }

	area = nation 					# Stays within the nation
	government = any	
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	gfx_type = culture_province
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	unit_transfer = yes
	will_relocate = yes
	
	infantry = 0.7
	cavalry = 0.3
	
	morale = 1.1
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no		
	handle_action_change_culture = yes
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1000
		modifier = {
			factor = 0
			owner = { NOT = { has_disaster = blood_lotus_rebellion } }
		}
		modifier = {
			factor = 0
			NOT = {
				culture_group = upper_raheni
				culture_group = middle_raheni
				culture_group = south_raheni
				has_human_minority_trigger = yes
			}
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
		modifier = {
			factor = 5
			fort_level = 2
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		always = yes
 	}
	siege_won_effect = {
		if = {
			limit = {
				superregion = rahen_superregion
				fort_level = 2
			}
			add_province_modifier = {
				name = blood_lotus_headquarter
				duration = -1
			}
			remove_province_modifier = blood_lotus_sympathizers
			remove_province_modifier = blood_lotus_guerillas
			remove_province_modifier = blood_lotus_guerillas_hidden
		}
		hidden_effect = { add_devastation = 10 }
		add_nationalism =  10
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "blood_lotus_rebellion_demand"
	
	# Country Scope
	demands_enforced_effect = {
		every_owned_province = {
			remove_province_modifier = blood_lotus_guerillas
			remove_province_modifier = blood_lotus_guerillas_hidden
			remove_province_modifier = blood_lotus_headquarter
			remove_province_modifier = blood_lotus_sympathizers
		}
		add_prestige = -100
		add_stability = -5
		hidden_effect = { set_country_flag = blood_lotus_lost }
	}
}
