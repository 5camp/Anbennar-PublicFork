#Triggered effects for the Deepwoods

hunt_lord = {
	hidden_effect = {
		random_list = {
			$pos_ch$ = { country_event = { id = $pos_out$ } }
			$neg_ch$ = { country_event = { id = $neg_out$ } }
		}
	}
}

hunt_lord_estate = {
	hidden_effect = {
		random_list = {
			$pos_ch$ = { country_event = { id = $pos_out$ } }
				modifier = {
					factor = 1.5
					has_country_flag = $companion$_on_hunt
				}
			$neg_ch$ = { country_event = { id = $neg_out$ } }
		}
	}
}
