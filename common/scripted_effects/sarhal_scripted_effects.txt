night_coven_collapse = {
    set_variable = {
        which = NightReformProv
        value = 1
    }
    every_owned_province = {
        NOT = {
            region = yezel_mora_region
        }
        remove_core = FROM
        random_core_country = {
            FROM = {
                Release = PREV
            }
        }
    }
    while = {
        limit = {
            all_owned_province = {
                NOT = {
                    region = yezel_mora_region
                }
            }
        }
        random_owned_province = {
            NOT = {
                region = yezel_mora_region
            }
            random_core_country = {
                export_to_variable = {
                    which = NightReformProv
                    value = trigger_value:tag
                    who = FROM
                }
            }
            cede_province = NightReformProv:FROM::tag
        }
    }
}
