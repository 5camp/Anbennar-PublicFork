cardesti_minority_migration_scripted_effect = {
	event_target:cardesti_province_$VAL$ = {
		remove_province_modifier = cardesti_minority_modifier
	}
	random_province = {
		limit = {
			continent = europe
			NOT = {
				has_province_modifier = cardesti_minority_modifier
				owner = {
					has_country_modifier = monstrous_nation
				}
			}
		}
		add_province_modifier = {
			name = cardesti_minority_modifier
			duration = -1
		}
		clear_global_event_target = cardesti_province_$VAL$
		save_global_event_target_as = cardesti_province_$VAL$
	}
}
