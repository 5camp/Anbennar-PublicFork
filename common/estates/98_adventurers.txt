estate_adventurers =
{
	icon = 10

	# If true, country will get estate
	trigger = {	
		NOT = { government = native }
		NOT = { has_reform = cossacks_reform }
		NOT = { has_reform = siberian_tribe }
		is_nomad = no
		
		#Anbennar
		NOT = { has_reform = adventurer_reform }
		
		NOT = { has_country_modifier = monstrous_nation }
	
	}

	# These scale with loyalty & power
	country_modifier_happy = {
		mercenary_cost = -0.2
		mercenary_discipline = 0.1
	}
	country_modifier_neutral = {
		mercenary_cost = -0.2
	}	
	country_modifier_angry = {
		mercenary_cost = 0.2
		mercenary_discipline = -0.1
	}
	land_ownership_modifier = {
		adventurers_loyalty_modifier = 0.2
	}
	
	province_independence_weight = {
		factor = 1
		modifier = {	#adventuring cities
			factor = 1.5
			development = 20
		}
		modifier = {	#territories where adventurers roam -> the frontier
			factor = 2
			NOT = { is_state_core = owner }
		}
	}
	
	# Influence modifiers
	base_influence = 10

	influence_modifier = {
		desc = EST_VAL_ADVENTURER_DERIVED_GOVERNMENT
		trigger = {
			OR = {
				has_reform = adventurer_kingdom_reform
				has_reform = adventurer_republic_reform
				has_country_flag = adventurer_derived_government
			}
		}
		influence = 10
	}
	
	loyalty_modifier = {
		desc = EST_VAL_MERITOCRATIC_RECRUITMENT_LOY
		trigger = {
			has_reform = meritocratic_focus_reform
		}
		loyalty = 5
	}
	loyalty_modifier = {
		desc = EST_VAL_ADVENTURER_DERIVED_GOVERNMENT
		trigger = {
			OR = {
				has_reform = adventurer_kingdom_reform
				has_reform = adventurer_republic_reform
			}
		}
		loyalty = 10
	}
	
	# Custom Names


	color = { 236 60 123 }

	privileges = {

	}

	agendas = {

	}
	influence_from_dev_modifier = 0.1
}
