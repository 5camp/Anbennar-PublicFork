great_kinstrife = {
	potential = {
		#NOT = { has_country_flag = had_cartel_wars }
		tag = I98
		OR = {
			is_year = 1730
			has_institution = enlightenment
			has_country_modifier = allclan_tension_warning
			has_country_modifier = allclan_tension_overflowing
		}
	}


	can_start = {
		OR = {
			has_institution = industrialization
			is_year = 1780
			has_disaster = allclan_council_disturbance
			if = { #this is to hide the tension modifier's existence if you don't have it
				limit = {
					has_country_modifier = allclan_tension_overflowing
				}
				has_country_modifier = allclan_tension_overflowing
			}
		}
	}
	
	
	can_stop = {
		always = no
		NOT = {
			has_disaster = allclan_council_disturbance
		}
		NOT = {
			has_institution = industrialization
			is_year = 1780
			if = { #this is to hide the tension modifier's existence if you don't have it
				limit = {
					has_country_modifier = allclan_tension_overflowing
				}
				has_country_modifier = allclan_tension_overflowing
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 1
			always = yes
		}

		modifier = {
			factor = 1

			OR = { 
				has_institution = industrialization
				is_year = 1780
				if = { #this is to hide the tension modifier's existence if you don't have it
					limit = {
						has_country_modifier = allclan_tension_overflowing
					}
					has_country_modifier = allclan_tension_overflowing
				}
			}
		}
	}


	can_end = {
		has_country_flag = ended_the_great_kinstrife
	}
	



	modifier = {
		land_morale = -0.1
		land_maintenance_modifier = 0.25
		stability_cost_modifier = 1
		global_unrest = 2
		global_autonomy = 0.05
		prestige_decay = 0.05
		republican_tradition = -2
	}
	
	on_start = allclan.800
	on_end = allclan.899
	
	on_monthly = {
		change_variable = {
			which = MonthsSinceKinstrife
			value = 1
		}
		events = {
			if = {
				limit = {
					check_variable = {
						which = MonthsSinceKinstrife
						value = 61 #more than 5 years since start
					}
				}
				random = {
					chance = 10
	
					country_event = {
						id = allclan.805 #orc uprising
						days = 0
					}
				}

				random = {
					chance = 10
	
					country_event = {
						id = allclan.811 #dwarf uprising
						days = 0
					}
				}
			}
		}
		random_events = {
			1000 = 0

			50 = allclan.851
			50 = allclan.852
			50 = allclan.853

			10 = allclan.805 #orc uprising

			10 = allclan.811 #dwarf uprising
			50 = allclan.812 #dwarf rebels
			50 = allclan.813 #dwarf rebels
			50 = allclan.814 #dwarf rebels quelled
		}
	}
}
