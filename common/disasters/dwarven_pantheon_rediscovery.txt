
dwarven_pantheon_rediscovery = {
	potential = {
		if = {
			limit = { religion = ancestor_worship }
			has_country_flag = dwarven_pantheon_in_rediscovery
		}
		else_if = {
			limit = { religion = dwarven_pantheon }
			NOT = { has_country_flag = dwarven_pantheon_rediscovery_completed }
		}
	}

	
	
	can_start = {
		always = yes
	}
	
	
	can_stop = {
        always = no
	}
	
	progress = {
		modifier = {
			factor = 0
			always = no
		}
		modifier = {
			factor = 20
			religion = dwarven_pantheon
		}
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = dwarven_pantheon_rediscovery_end_tooltip
			always = no
		}
	}
	
	modifier = {
		global_unrest = 6
		global_autonomy = 0.3
		stability_cost_modifier = 1
		all_estate_loyalty_equilibrium = -0.1
		missionaries = 1
		global_heretic_missionary_strength = 0.02
	}
	
	on_start = dwarven_pantheon.200
	on_end = flavor_krak.99
	
	on_monthly = {
		events = {}
		random_events = {
			1060 = 0
			130 = flavor_krak.66 #Peasant revolt
			90 = flavor_krak.67 #worker strike
			50 = flavor_krak.65 #noble fort
		}
	}
}
