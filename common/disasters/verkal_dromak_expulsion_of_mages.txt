verkal_dromak_expulsion_of_mages = {
	potential = {
		has_country_flag = verkal_dromak_mage_expulsion
		NOT = { has_country_flag = verkal_dromak_mage_expulsion_happened }
	}

	
	
	can_start = {
		always = yes
	}
	
	
	can_stop = {
        always = no
	}
	
	progress = {
		modifier = {
			factor = 100
			custom_trigger_tooltip = {
				tooltip = something_is_going_up_tooltip
				always = yes
			}
		}
	}
	
	can_end = {
		OR = {
			#Option 1: Diplomatic way
			custom_trigger_tooltip = {
				tooltip = verkal_dromak_mages_expelled
				has_country_flag = banned_magic
			}
			
			#Option 2: Killed all mages
			custom_trigger_tooltip = {
				tooltip = verkal_dromak_mages_lost
				estate_influence = {
					estate = estate_mages
					influence = 0
				}
			}
		}
	}
	
	modifier = {
		land_morale = -0.1
		global_manpower = -0.2
		all_power_cost = 0.2
	}
	
	on_start = verkaldromak.175
	on_end = verkaldromak.8
	
	on_monthly = {
		events = {
		}
		random_events = {
			100 = verkaldromak.170 #kill mages in hold
			20 = verkaldromak.171 #destroy mage tower
			100 = verkaldromak.172 #sweep through the rails
			200 = verkaldromak.173 #spawn rebels
			1000 = 0
		}
	}
}
