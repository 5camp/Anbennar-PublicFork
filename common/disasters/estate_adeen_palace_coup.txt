
estate_adeen_palace_coup = {
	potential = {
		has_estate = estate_adeen
		culture_group = baashidi
		num_of_cities = 5
		check_variable = {
			adeen_appease_count = 10
		}
		NOT = {
			OR = {
				has_country_flag = adeen_coup_won
				has_country_flag = adeen_coup_lost
				has_country_flag = adeen_integrated
			}
		}
	}
	
	
	can_start = {
		has_any_disaster = no
		#NOT = { legitimacy_equivalent = 75 }
		custom_trigger_tooltip = {
			tooltip = adeen_disaster_can_start
			check_variable = {
				adeen_appease_count = 15
			}
		}
		NOT = { has_estate_privilege = estate_adeen_power_over_the_throne }
		OR = {
			NOT = { stability = 0 }
			OR = {
				estate_influence = {
					estate = estate_adeen
					influence = 50
				}
				AND = {
					OR = {
						has_estate_privilege = estate_adeen_focus_on_harpies
						has_estate_privilege = estate_adeen_focus_on_lizardfolk
					}
					NOT = {
						estate_loyalty = {
							estate = estate_adeen
							loyalty = 50
						}
					}
				}
			}
		}
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			AND = {
				legitimacy_equivalent = 75
				OR = {
					NOT = {
						estate_influence = {
							estate = estate_adeen
							influence = 30
						}
					}
				}
			}
			has_estate_privilege = estate_adeen_power_over_the_throne
		}
	}
		
	
	
	progress = {
		modifier = {
			factor = 5
			always = yes
		}	
	}
	
	
	
	can_end = {
		stability = 0
		NOT = { num_of_rebel_armies = 1 }
		NOT = { num_of_rebel_controlled_provinces = 1 }
	}
	

	modifier = {
		global_unrest = 3
		stability_cost_modifier = 0.5
	}	
	
	
	on_start = estate_adeen.100
	on_end = estate_adeen.101
	
	on_monthly = {
		random_events = {
			150 = 0
			10 = flavour_gor_burad.51
		}
	}
}
