fallen_dwarf = {
	potential = {
		has_country_flag = fallen_disaster_start
	}

	
	
	can_start = {
		always = yes
	}
	
	
	can_stop = {
        always = no
	}
	
	progress = {
		modifier = {
			factor = 1
			always = yes
		}
	}
	
	
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = no_old_hold_tooltip
			all_owned_province = {
				OR = {
					has_province_modifier = old_hold
					has_province_modifier = old_gate
				}
				controlled_by = ROOT
			}
		}
		custom_trigger_tooltip = {
			tooltip = fight_great_invasion_tooltip
			has_country_flag = fallen_disaster_end
		}
		custom_trigger_tooltip = {
			tooltip = destroy_old_hold_tooltip
			all_owned_province = {
				NOT = { has_province_modifier = old_hold }
			}
		}
		NOT = { has_spawned_rebels = fallen_dwarf_rebel }
	}
	

	modifier = {
		global_unrest = 3
		global_colonial_growth = -100
	}
	
	
	on_start = diggy_fallen.1
	on_end = aelnar.18
	
	on_monthly = {
		events = {
		}
		random_events = { 
			1000 = 0
			200 = diggy_fallen.2 #Attack in old hold
			100 = diggy_fallen.3 #Attack from gate
		
		}
	}
}
