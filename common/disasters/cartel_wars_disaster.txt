cartel_wars = {
	potential = {
		NOT = { has_country_flag = had_cartel_wars }
		culture_group = goblin
		is_free_or_tributary_trigger = yes
		OR = {
			current_age = age_of_absolutism
		}
		num_of_cities = 15
	}


	can_start = {
		current_age = age_of_absolutism
		has_any_disaster = no
		NOT = {
			absolutism = 100
		}
		OR = {
			yearly_corruption_increase = 0.001
			corruption = 5
			monthly_income = 200
			has_global_modifier_value = {
				which = global_tax_modifier
				value = 0.2
			}
			has_global_modifier_value = {
				which = production_efficiency
				value = 0.2
			}
			has_global_modifier_value = {
				which = global_trade_goods_size_modifier
				value = 0.2
			}
			has_global_modifier_value = {
				which = global_trade_power
				value = 0.2
			}
			has_global_modifier_value = {
				which = trade_efficiency
				value = 0.2
			}
		}
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			#stability = 3
			absolutism = 100
		}
	}
	
	progress = {
		modifier = {
			factor = 1
			corruption = 5
		}
		modifier = {
			factor = 0.5
			unrest = 1
		}
		modifier = {
			factor = 1
			yearly_corruption_increase = 0.001
		}
		modifier = {
			factor = 1
			NOT = { absolutism = 20 }
		}
		modifier = {
			factor = 1
			trade_income_percentage = 0.51
		}
		modifier = {
			factor = 1
			NOT = { absolutism = 50 }
		}
		modifier = {
			factor = -1
			NOT = { unrest = 1 }
		}
	}




	
	can_end = {
		NOT = { num_of_rebel_controlled_provinces = 1 }
		stability = 0
		NOT = { war_exhaustion = 2 }
		custom_trigger_tooltip = {
			tooltip = disaster_active_for_10_years_tooltip
			had_country_flag = { flag = had_court_and_country days = 3650 }
		}
	}
	



	modifier = {
		global_trade_goods_size_modifier = -0.2
		global_tax_modifier = -0.2
		max_absolutism = -20
	}
	
	on_start = court_and_country_events.1
	on_end = court_and_country_events.100
	
	on_monthly = {
		events = {
		}

			1000 = 0
			50 = court_and_country_events.2 #
			50 = court_and_country_events.3 #

			50 = court_and_country_events.5 #

		}
	}
