cartel_wars = {
	potential = {
		NOT = { has_country_flag = had_cartel_wars }
		culture_group = goblin
		is_free_or_tributary_trigger = yes
		OR = {
			current_age = age_of_absolutism
		}
		num_of_cities = 15
		#has_estate = estate_burghers
	}


	can_start = {
		current_age = age_of_absolutism
		has_any_disaster = no
		NOT = {
			absolutism = 100
		}
		OR = {
			yearly_corruption_increase = 0.001
			corruption = 5
			stability = -3
			monthly_income = 200
			has_global_modifier_value = {
				which = production_efficiency
				value = 0.2
			}
			has_global_modifier_value = {
				which = global_trade_goods_size_modifier
				value = 0.2
			}
			has_global_modifier_value = {
				which = global_trade_power
				value = 0.2
			}
			has_global_modifier_value = {
				which = trade_efficiency
				value = 0.2
			}
		}
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			#stability = 3
			absolutism = 100
		}
	}
	
	progress = {
		modifier = {
			factor = 0.1
			corruption = 5
		}
		modifier = {
			factor = 0.1
			unrest = 1
		}
		modifier = {
			factor = 0.1
			yearly_corruption_increase = 0.001
		}
		modifier = {
			factor = 0.1
			NOT = { absolutism = 20 }
		}
		modifier = {
			factor = 0.5
			trade_income_percentage = 0.51
		}
		modifier = {
			factor = 0.1
			has_global_modifier_value = {
				which = production_efficiency
				value = 0.2
			}
		}
		modifier = {
			factor = 0.1
			has_global_modifier_value = {
				which = trade_efficiency
				value = 0.2
			}
		}
		modifier = {
			factor = 0.1
			has_global_modifier_value = {
				which = global_trade_power
				value = 0.2
			}
		}
		modifier = {
			factor = 0.1
			has_global_modifier_value = {
				which = global_trade_goods_size_modifier
				value = 0.2
			}
		}
	}


	can_end = {
		#NOT = { num_of_rebel_controlled_provinces = 1 }
		#stability = 0
		OR = {
			has_country_flag = end_of_the_cartel_wars
			has_country_flag = cartels_defeated
			has_country_flag = cartels_accomodated
			has_country_flag = anti-cartel_rebels_won
		}
	}
	



	modifier = {
		yearly_corruption = 0.05
		yearly_absolutism = -1
		global_trade_goods_size_modifier = 0.2 #cartels overproducing
		burgher_influence_modifier = 0.1
		#burgher_loyalty_modifier = -0.1
	}
	
	on_start = cartel_wars_events.1
	on_end = cartel_wars_events.999
	
	on_monthly = {
		events = {
		}
		random_events = {
			1000 = 0
			
			#cartels grow in power. Stab hit, lose 20 legitimacy/10 rt and gain 5 corruption each time the tier goes up
			50 = cartel_wars_events.2 #hidden event that fires the others. autofires after 2 years
			#3 level 1 +corruption, -gov capacity, -max absolutism, +stab cost, and minus prod and trade efficiency per tier, alongside +goods produced and reduced burgher loyalty with increased influence
			#4 level 2
			#5 level 3
			#6 level 4
			#7 level 5
			#8 stab loss, 10 corruption, and -10 legitmacy/rt if already at level 5

			#events below enabled by level 1 cartel influence
			#cartels affect the nation
			25 = cartel_wars_events.9 #smuggling causes massive loss of trade power
			25 = cartel_wars_events.10 #cartels charging their own tariffs gives loss of diplomacy
			25 = cartel_wars_events.11 #cartels ruining the army
			25 = cartel_wars_events.12 #cartels ruin the administration raising advisor and state maintenance costs
			25 = cartel_wars_events.13 #cartels take over your finances, meaning through the roof inflation and interest rates

			#gang fights. They get more severe at higher levels of cartel influence
			50 = cartel_wars_events.14 #hidden event to fire gang wars. Has chance to do nothing that goes away at level 3 cartel influence
			50 = cartel_wars_events.15 #gang fight in the capital
			50 = cartel_wars_events.16 #minor gang fight in one of your provinces
			50 = cartel_wars_events.17 #gang fight in a center of trade
			50 = cartel_wars_events.18 #gang fight in a foreign nation
			50 = cartel_wars_events.19 #mega gang fight in one of your provinces

			#assassinations enabled by level 4 cartel influence
			100 = cartel_wars_events.20 #hidden event. Has 50% chance to do nothing until level 5
			25 = cartel_wars_events.21 #assassination of a general
			25 = cartel_wars_events.22 #assassination of an advisor
			25 = cartel_wars_events.23 #assassination of a foreign dignitary
			25 = cartel_wars_events.24 #assassination of a minor cartel boss
			25 = cartel_wars_events.25 #assassination of local official
			25 = cartel_wars_events.26 #assassination of major cartel boss
			#next three only possible at level 5
			25 = cartel_wars_events.27 #assassination of your ruler
			25 = cartel_wars_events.28 #assassination of your heir
			25 = cartel_wars_events.29 #assassination of your consort
		}
	}
}
