leechfather_reforms = {
    trigger = {
        religion = leechfather
    }    

    can_buy_idea = {
		NOT = { doom = 50 }
		any_known_country = {
			is_in_war = {
			  defender_leader = THIS
			  attackers = ROOT
			}
			capital_scope = {
				controlled_by = ROOT
			}
			OR = {
				current_war_losses = {
					type = men
					cause = battles
					amount = 1
				}
				AND = {
					NOT = { 
						manpower = 0.01
					}
					current_war_losses = {
						type = men
						cause = battles
						amount = 1
					}
				}
			}
		}
    }
    
    leechfather_reform_1 = {
        global_monthly_devastation = -0.15
    }
    leechfather_reform_2 = {
        free_mil_policy = 1
    }
    leechfather_reform_3 = {
        global_sailors_modifier = 0.15
    }
    leechfather_reform_4 = {
        culture_conversion_cost = -0.25
    }
    leechfather_reform_5 = {
        colonists = 1
    }     
    
    ai_will_do = {
        factor = 1
    }
}
