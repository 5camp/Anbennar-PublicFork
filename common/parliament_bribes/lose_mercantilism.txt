lose_mercantilism = {
	trigger = {
		has_reached_seat_threshold = no
		mercantilism = 20
		has_not_bribe_blocking_debate_active = {
			type = mercantilism
		}
	}
	
	effect = {
		if = {
			limit = {
				has_expensive_debate = yes
				NOT = { halved_parliament_bribes = yes }
			}
			add_mercantilism = -4
		}
		else_if = {
			limit = {
				OR = {
					AND = {
						has_expensive_debate = yes
						halved_parliament_bribes = yes
					}
					AND = {
						NOT = { has_expensive_debate = yes }
						NOT = { halved_parliament_bribes = yes }
					}
				}
			}
			add_mercantilism = -2
		}
		else = {
			add_mercantilism = -1
		}
		if = {
			limit = {
				OR = { 
					has_reform = azkare_court
					has_reform = sunrise_court_reform
				}
			}
			every_owned_province = {
				limit = { 
					has_seat_in_parliament = yes
					current_bribe = lose_mercantilism
					NOT = { province_id = FROM }
				}
				back_current_issue = yes
				add_local_autonomy = 20
			}
		}
	}
	
	chance = {
		factor = 1
	}

	ai_will_do = {
		factor = 1
	}	
}
