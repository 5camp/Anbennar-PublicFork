artifice_invention_basic_template = {
	icon = privilege_promote_maratha_nobility
	max_absolutism = 0
	loyalty = 0
	influence = 0
	is_valid = {
		always = no
	}
	can_select = {
		artifice_has_points_for_basic = yes
	}
	can_revoke = {
		always = yes
	}
	on_granted = {
		artifice_granted_basic = yes
		#update_artifice_points = yes #is included in the above effect

		#for illustration of a possible effect in missions
		mission_reward_artifice_points_small = yes
	}
	on_revoked = {
		artifice_remove_basic = yes
		#update_artifice_points = yes #is included in the above effect
	}
	penalties = {
		#foo
	}
	benefits = {
		#foo
	}
	ai_will_do = {
		factor = 5
	}
}
