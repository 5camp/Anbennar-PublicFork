estate_shinukhorchi_monopoly_of_damestear = {
	icon = privilege_monopoly_of_goods_damestear
	max_absolutism = -10
	loyalty = 0.1
	is_valid = {
		has_dlc = "Emperor"
		any_owned_province = {
			trade_goods = damestear
		}
	}
	can_select = {
		custom_trigger_tooltip = {
			tooltip = unavailable_during_revolution_disaster
			NOT = { has_disaster = revolution }
			NOT = {
				has_disaster_progress = {
					disaster = revolution
					value = 1
				}
			}
		}
	}
	on_granted = {
		custom_tooltip = estate_shirgrii_increasing_privilege_limit_tooltip
		add_mercantilism = 1
		add_years_of_owned_provinces_production_income = {
			years = 8
			trigger = {
				trade_goods = damestear
			}
			custom_tooltip = MONOPOLY_INCOME_DAMESTEAR_DESC
		}
		custom_tooltip = estate_mages_monopoly_of_damestear_effect_tooltip
		hidden_effect = {
			change_variable = {
				which = var_oni_magic_privilege_limit
				value = 1
			}
		}
	}
	on_granted_province = {
		if = {
			limit = {
				trade_goods = damestear
			}
			add_province_modifier = {
				name = estate_shinukhorchi_monopoly_of_damestear_mod
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = estate_shinukhorchi_monopoly_of_damestear_mod
			}
			remove_province_modifier = estate_shinukhorchi_monopoly_of_damestear_mod
		}
	}
	on_revoked = {
		custom_tooltip = estate_mages_monopoly_of_damestear_revoke_tooltip
		hidden_effect = {
			change_variable = {
				which = var_oni_magic_privilege_limit
				value = -1
			}
		}
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = estate_shinukhorchi_monopoly_of_damestear_mod
			}
			remove_province_modifier = estate_shinukhorchi_monopoly_of_damestear_mod
		}
	}
	on_invalid = {
		custom_tooltip = estate_mages_monopoly_of_damestear_revoke_tooltip
		hidden_effect = {
			change_variable = {
				which = var_oni_magic_privilege_limit
				value = -1
			}
		}
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = estate_shinukhorchi_monopoly_of_damestear_mod
			}
			remove_province_modifier = estate_shinukhorchi_monopoly_of_damestear_mod
		}
	}
	penalties = {
	}
	benefits = {
	}

		country_event = { id = monopoly_privileges.12 }
	}

	}
