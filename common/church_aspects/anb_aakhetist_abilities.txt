aakhetist_offer_servitude = {
	sprite = "GFX_look_for_omens"
	cost = 25

		prestige = 15
	}

	effect = {
		custom_tooltip = aakhetist_offer_servitude_tooltip
		#custom_tooltip = aakhetist_current_approval_tooltip
		hidden_effect = { country_event = { id = aakhetist.20 } }
	}
}

aakhetist_offer_food = {
	sprite = "GFX_invoke_warrior_heritage"
	cost = 50
	
	trigger = {
		any_owned_province = {
			OR = {
				trade_goods = grain
				trade_goods = livestock
			}
			development = 5
		}
	}

	effect = {
		custom_tooltip = aakhetist_offer_food_tooltip
		#custom_tooltip = aakhetist_current_approval_tooltip
		hidden_effect = { country_event = { id = aakhetist.21 } }
	}
}

aakhetist_offer_weapons = {
	sprite = "GFX_assemble_shamans_old"
	cost = 50
	
	trigger = {
		if = {
			limit = { total_development = 1000 }
			treasury = 1000
		}
		else_if = {
			limit = { total_development = 750 }
			treasury = 750
		}
		else_if = {
			limit = { total_development = 500 }
			treasury = 500
		}
		else = { years_of_income = 1.5 }
	}

	effect = {
		custom_tooltip = aakhetist_offer_weapons_tooltip
		custom_tooltip = aakhetist_current_approval_tooltip
		hidden_effect = { country_event = { id = aakhetist.22 } }
	}
}

denounce_yrilaks_followers = {
	sprite = "GFX_diplomats_of_the_reformation_aspect"
	cost = 100
	
	trigger = {
		any_known_country = {
			is_enemy = ROOT
			custom_trigger_tooltip = {
				tooltip = denounce_yrilak_follower_trigger_tt
				OR = {
					ruler_has_bad_adm_personality = yes
					ruler_has_bad_dip_personality = yes
					ruler_has_bad_mil_personality = yes
				}
			}
		}
		diplomatic_reputation = 2
		NOT = { coalition_target = ROOT }
		estate_loyalty = {
			estate = estate_church
			loyalty = 60
		}
		NOT = {
			has_country_modifier = against_yrilakites
		}
	}

	potential = {
	}

	effect = {
		country_event = {
			id = sky_domain.6
		}
	}

}
