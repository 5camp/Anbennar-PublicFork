pillage_and_burn_aspect {
	sprite = "GFX_dissolve_monasteries_icon"
	cost = 20
	
	trigger = {
		is_at_war = yes
		any_province = {
			controlled_by = ROOT
			NOT = { owned_by = ROOT }
		}
	}
	
	effect = {
		random_province = {
			limit = {
				controlled_by = ROOT
				NOT = { owned_by = ROOT}
			}
			random_list = {
				1 = { add_base_tax = -1 }
				1 = { add_base_production = -1 }
				1 = { add_base_manpower = -1 }
			}
			add_devastation = 80
		}
		
		capital_scope {
			random_list = {
				1 = { add_base_tax = 1 }
				1 = { add_base_production = 1 }
				1 = { add_base_manpower = 1 }
			}
		}
	}
	
	ai_will_do = {
		factor = 1
	}
}

purge_the_weak_aspect {
	sprite = "GFX_dissolve_monasteries_icon"
	cost = 100
	
	trigger = {
		is_at_war = no
		any_owned_province = {
			OR = {
				has_owner_religion = no
				has_owner_religion = no
			}
		}
	}
	
	effect = {
		random_owned_province = {
			limit = {
				OR = {
					has_owner_religion = no
					has_owner_religion = no
				}
			}
			add_devastation = 50
			create_revolt = 1
			change_culture = ROOT
			change_religion = ROOT
		}
	}
	
	ai_will_do = {
		factor = 1
	}
}
