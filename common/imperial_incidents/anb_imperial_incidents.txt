incident_halfling_elector = {
	event = flavor_beepeck.2
	default_option = 1
	
	can_stop = {
		OR = {
			NOT = { exists = A12 }
			A12 = { is_part_of_hre = no }
			A12 = { is_elector = no }
		}
	}
	
	0 = { # reject claims
		factor = 1
		modifier = {
			factor = 5
			OR = {
				ai_attitude = {
					who = A12
					attitude = attitude_rivalry
				}
				ai_attitude = {
					who = A12
					attitude = attitude_hostile
				}
				ai_attitude = {
					who = A12
					attitude = attitude_threatened
				}
				ai_attitude = {
					who = A12
					attitude = attitude_outraged
				}
				ai_attitude = {
					who = A12
					attitude = attitude_rebellious
				}
			}
		}
		modifier = {
			factor = 3
			NOT = {
				has_opinion = {
					who = A12
					value = -4
				}
			}
		}
		modifier = {
			factor = 0
			tag = A12
		}
	}
	
	1 = { # acknowledge claim
		factor = 1
		modifier = {
			factor = 2
			OR = {
				ai_attitude = {
					who = A12
					attitude = attitude_disloyal
				}
				has_opinion = {
					who = A12
					value = 15
				}
			}
			NOT = {
				has_opinion = {
					who = A12
					value = 35
				}
			}
		}
		modifier = {
			factor = 0
			owns = 157
		}
	}
	
	2 = { # enforce claim.
		factor = 1
		modifier = {
			factor = 3
			OR = {
				has_opinion = {
					who = A12
					value = 35
				}				
				ai_attitude = {
					who = A12
					attitude = attitude_friendly
				}
				ai_attitude = {
					who = A12
					attitude = attitude_loyal
				}
				ai_attitude = {
					who = A12
					attitude = attitude_allied
				}
			}
		}
		modifier = {
			factor = 10
			tag = A12
		}
		modifier = {
			factor = 0
			owns = 157
		}
	}
}

incident_witch_prince = {
	event = witch_king.4
	default_option = 0
	
	can_stop = {
		OR = {
			NOT = { exists = event_target:imperial_witch_king }
			event_target:imperial_witch_king = { is_part_of_hre = no }
			event_target:imperial_witch_king = {
				is_not_evil = yes
			}
		}
	}
	
	#They have forfeited their right
	0 = {
		factor = 10
		#If you don't like the Witch King
		modifier = {
			factor = 5
			OR = {
				ai_attitude = {
					who = event_target:imperial_witch_king
					attitude = attitude_rivalry
				}
				ai_attitude = {
					who = event_target:imperial_witch_king
					attitude = attitude_hostile
				}
				ai_attitude = {
					who = event_target:imperial_witch_king
					attitude = attitude_threatened
				}
				ai_attitude = {
					who = event_target:imperial_witch_king
					attitude = attitude_outraged
				}
				ai_attitude = {
					who = event_target:imperial_witch_king
					attitude = attitude_rebellious
				}
				is_neighbor_of = event_target:imperial_witch_king
			}
		}
		#Magical Authorities
		modifier = {
			factor = 10
			OR = {
				tag = A85 #Magisterium
				tag = A77 #Orda Aldresia
				is_emperor = yes
			}
			NOT = {
				alliance_with = event_target:imperial_witch_king
			}
		}
		#If you are the Witch King
		modifier = {
			factor = 0
			tag = event_target:imperial_witch_king
		}
	}
	
	#Denounce them
	1 = {
		factor = 5
		#If you like the Witch King
		modifier = {
			factor = 10
			OR = {
				has_opinion = {
					who = event_target:imperial_witch_king
					value = 100
				}				
				ai_attitude = {
					who = event_target:imperial_witch_king
					attitude = attitude_friendly
				}
				ai_attitude = {
					who = event_target:imperial_witch_king
					attitude = attitude_loyal
				}
				ai_attitude = {
					who = event_target:imperial_witch_king
					attitude = attitude_allied
				}
			}
		}
		#If you don't like the emperor
		modifier = {
			factor = 5
			OR = {
				ai_attitude = {
					who = emperor
					attitude = attitude_rivalry
				}
				ai_attitude = {
					who = emperor
					attitude = attitude_hostile
				}
				ai_attitude = {
					who = emperor
					attitude = attitude_threatened
				}
				ai_attitude = {
					who = emperor
					attitude = attitude_outraged
				}
				ai_attitude = {
					who = emperor
					attitude = attitude_rebellious
				}
			}
		}
	}
}
