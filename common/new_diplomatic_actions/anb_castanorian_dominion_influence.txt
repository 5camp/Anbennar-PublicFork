# QOL thing that gives your claims in the dominion's capital region to the dominion
castanor_give_capital_claims = {
	category = influence
	
	is_visible = {
		tag = B32
		FROM = {
			is_subject_of_type = castanor_dominion
			is_subject_of = ROOT
		}
    }
	
	is_allowed = {
		hidden_trigger = {
			tag = B32
			FROM = {
				is_subject_of_type = castanor_dominion
				is_subject_of = ROOT
			}
		}
	}

    on_accept = {
		FROM = {
			capital_scope = {
				region = {
					if = {
						limit = { OR = { is_claim = ROOT NOT = { is_permanent_claim = ROOT } } }
						add_claim = FROM
						remove_claim = ROOT
					}
					else_if = {
						limit = { is_permanent_claim = ROOT }
						add_permanent_claim = FROM
						remove_claim = ROOT
					}
				}
			}
		}
	}
	ai_will_do = {
		always = no
	}
}
# Encourage assimilation/promote Castanorian culture 
castanor_encourage_assimilation = {
	category = influence
	
	is_visible = {
		tag = B32
		FROM = {
			is_subject_of_type = castanor_dominion
			is_subject_of = ROOT
		}
    }
	
	is_allowed = {
		hidden_trigger = {
			tag = B32
			FROM = {
				is_subject_of_type = castanor_dominion
				is_subject_of = ROOT
			}
		}
		FROM = { NOT = { has_country_modifier = castanor_encourage_assimilation_mod_dominion } }
		NOT = { has_country_modifier = castanor_encourage_assimilation_mod_castanor }
	}

    on_accept = {
		FROM = {
			add_country_modifier = {
				name = castanor_encourage_assimilation_mod_dominion
				duration = 7300
			}
		}
		add_splendor = -400
		add_country_modifier = {
			name = castanor_encourage_assimilation_mod_castanor
			duration = 7300
		}
	}
	ai_will_do = {
		always = no
	}
}
# Press auxilaries 
castanor_auxilaries = {
	category = influence
	
	is_visible = {
		tag = B32
		FROM = {
			is_subject_of_type = castanor_dominion
			is_subject_of = ROOT
		}
    }
	
	is_allowed = {
		hidden_trigger = {
			tag = B32
			FROM = {
				is_subject_of_type = castanor_dominion
				is_subject_of = ROOT
			}
		}
		FROM = {
			NOT = { has_country_modifier = castanor_auxilaries_mod_dominion }
			manpower = 5
		}
	}

    on_accept = {
		FROM = {
			add_country_modifier = {
				name = castanor_auxilaries_mod_dominion
				duration = 730
			}
			add_manpower = -10
		}
		add_splendor = -100
		add_manpower = 10
		capital_scope = {
			add_or_extend_province_modifier_effect = {
				MODIFIER = castanor_auxilaries_mod_castanor
				DURATION = 730
			}
		}
	}
	ai_will_do = {
		always = no
	}
}
# Integrate 
castanor_integrate_dominion = {
	category = influence
	
	is_visible = {
		tag = B32
		FROM = {
			is_subject_of_type = castanor_dominion
			is_subject_of = ROOT
		}
    }
	
	is_allowed = {
		hidden_trigger = {
			tag = B32
			FROM = {
				is_subject_of_type = castanor_dominion
				is_subject_of = ROOT
			}
		}
		FROM = {
			all_owned_province = {
				if = {
					limit = { has_human_majority_trigger = yes }
					culture = castanorian
				}
				else = { always = yes }
			}
		}
	}

    on_accept = {
		add_splendor = -800
		add_country_modifier = {
			name = castanor_integrate_dominion
			duration = 3650
		}
		inherit = FROM
	}
	ai_will_do = {
		always = no
	}
}
