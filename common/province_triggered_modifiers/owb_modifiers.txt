owb_ancient_precursor_artefact = {
	potential = {
		always = yes
	}
	
	hidden = yes

	trigger = {
		OR = {
			owner = {
				OR = {
					tag = OWB
					is_subject_of = OWB
				}
			}
			controlled_by = OWB
		}
	}
	
	on_activation = {
		OWB = {
			country_event = { id = owb.2 }
		}
		remove_province_triggered_modifier = owb_ancient_precursor_artefact
	}

	
	on_deactivation = {
	}
}

owb_precursor_fragment = {
	potential = {
		always = yes
	}
	
	# hidden = yes

	trigger = {
		owner = {
			OR = {
				tag = OWB
				is_subject_of = OWB
			}
		}
	}
	
	on_activation = {
		save_global_event_target_as = owb_fragments_global
		if = {
			limit = { has_province_flag = owb_glowing }
			OWB = { event_target:owb_fragments_global = { province_event = { id = owb.106 } } }
		}
		else_if = {
			limit = { has_province_flag = owb_dragon_house }
			OWB = { event_target:owb_fragments_global = { province_event = { id = owb.206 } } }
		}
		else_if = {
			limit = { has_province_flag = owb_elfenbride }
			OWB = { event_target:owb_fragments_global = { province_event = { id = owb.306 } } }
		}
		else_if = {
			limit = { has_province_flag = owb_god_fragment }
			OWB = { event_target:owb_fragments_global = { province_event = { id = owb.406 } } }
		}
		else_if = {
			limit = { has_province_flag = owb_vault }
			OWB = { event_target:owb_fragments_global = { province_event = { id = owb.506 } } }
		}
		else_if = {
			limit = { has_province_flag = owb_portal }
			OWB = { event_target:owb_fragments_global = { province_event = { id = owb.606 } } }
		}
		else_if = {
			limit = { has_province_flag = owb_floating }
			OWB = { event_target:owb_fragments_global = { province_event = { id = owb.706 } } }
		}
		remove_province_triggered_modifier = owb_precursor_fragment
	}

	
	on_deactivation = {
	}
}
