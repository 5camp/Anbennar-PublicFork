# All these triggers handle culture conversion in Aelantir
# This is the handler where you pick what to convert, what to convert to, and where to convert
aelantir_conversion_req = {
	[[native]
		has_owner_culture = no
		NOT = { current_age = age_of_discovery }
		NOT = { current_age = age_of_reformation }
		NOT = { has_province_flag = aelantir_culture_change }
		culture = $from_culture$
		owner = {
			is_colonial_nation = no
			OR = {
				capital_scope = {
					continent = north_america
				}
				capital_scope = {
					continent = south_america
				}
			}
		}
	[[colonialone]
		has_owner_culture = no
		NOT = { current_age = age_of_discovery }
		NOT = { current_age = age_of_reformation }
		NOT = { has_province_flag = aelantir_culture_change }
		culture = $from_culture$
		is_core = $core1$
		owner = {
			is_colonial_nation = yes
			OR = {
				capital_scope = {
					continent = north_america
				}
				capital_scope = {
					continent = south_america
				}
			}
		}
	]
	[[colonialtwo]
		has_owner_culture = no
		NOT = { current_age = age_of_discovery }
		NOT = { current_age = age_of_reformation }
		NOT = { has_province_flag = aelantir_culture_change }
		culture = $from_culture$
		OR {
			is_core = $core1$
			is_core = $core2$
		}
		owner = {
			is_colonial_nation = yes
			OR = {
				capital_scope = {
					continent = north_america
				}
				capital_scope = {
					continent = south_america
				}
			}
		}
	]
}

	ROOT = {
		OR = {
			[[onecore]
				colonial_region = colonial_haraf
				is_core = $core1$
			]
			[[twocore]
				colonial_region = colonial_haraf
				is_core = $core1$
				is_core = $core2$
			]
			[[nocore]
				colonial_region = colonial_haraf
			]
		}
	}
}
