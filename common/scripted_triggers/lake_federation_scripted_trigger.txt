is_lake_federation_leader = {
	custom_trigger_tooltip = {
		tooltip = is_federation_leader_tooltip
		
		NOT = {
			calc_true_if = {
				all_known_country = {
					check_variable = {
						which = FederationPoints
						which = PREV
					}
					is_subject = no
				}
				amount = 1
			}
		}
		is_subject = no
	}
}
