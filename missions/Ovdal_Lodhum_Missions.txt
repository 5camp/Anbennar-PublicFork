	}

	lodhum_feywoods = {
		icon = mission_alone_in_the_wood
		position = 5
		required_missions = {
			lodhum_wooden_door
		}

		provinces_to_highlight = {
			OR = {
				#just highlight ayeth
				AND = {
					exists = I33
					owned_by = I33
				}
				#stuff you should conquer
				AND = {
					NOT = { exists = I33 }
					is_core = ROOT
					NOT = { owned_by = ROOT }
					superregion = deepwoods_superregion
				}
			}
		}

		trigger = {
			if = {
				limit = { exists = I33 }
				has_opinion = { who = ROOT value = 100 }
				treasury = 100
				manpower = 2
			}
			else = {
				custom_trigger_tooltip = { #need at least 3, more provinces = more rewards
					tooltip = lodhum_feywoods_tooltip
					num_of_owned_provinces_with = {
						value = 3
						superregion = deepwoods_superregion
						owned_by = ROOT
						is_core = ROOT
						has_province_flag = lodhum_release_this
					}
				}
			}
		}
		
		effect = {
			if = {
				limit = { exists = I33 }
				add_treasury = -100
				add_manpower = -2
				I33 = {
					add_treasury = 100
					add_manpower = 2
				}
			}
			else = {
				custom_tooltip = lodhum_feywoods_tooltip2
				hidden_effect = {
					every_owned_province = {
						limit = {
							superregion = deepwoods_superregion
							is_core = ROOT
							has_province_flag = lodhum_release_this
						}
						ROOT = { 
							add_adm_power = 20
							add_treasury = 15
						}
						cede_province = I33
					}
					#seperate so it also removes stuff you didn't conquer
					every_core_province = { 
						limit = { superregion = deepwoods_superregion }
						remove_core = ROOT
					}
				}
				create_guarantee = I33
			}
			deepwoods_superregion = { discover_country = ROOT }
		}
	}
