country_decisions = {
	
	human_mages_toggle = {
		effect = {
			if = {
				limit = {
					has_country_flag = human_mages
				}
				clr_country_flag = human_mages
			}
			else = {
				set_country_flag = human_mages
			}
		}
	}
	
	harimari_mages_toggle = {
		effect = {
			if = {
				limit = {
					has_country_flag = harimari_mages
				}
				clr_country_flag = harimari_mages
			}
			else = {
				set_country_flag = harimari_mages
			}
		}
	}
}
